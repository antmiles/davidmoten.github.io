3   import java.io.File;
4   import java.io.IOException;
5   import java.nio.charset.StandardCharsets;
6   import java.nio.file.Files;
7   
8   /**
9    * <a href=
10   * "https://stackoverflow.com/questions/7918060/how-do-i-sort-very-large-files">question</a>
11   */
12  public class SO_HowDoISortVeryLargeFiles {
13  
14      public static void main(String[] args) throws IOException {
15  
16          String s = "0052304 0000004000000000000000000000000000000041   John Teddy   000023\n"
17                  + "0022024 0000004000000000000000000000000000000041   George Clan 00013";
18          File input = new File("target/input");
19          Files.write(input.toPath(), s.getBytes(StandardCharsets.UTF_8));
20          File output = new File("target/output");
21  
22          Sorter.serializerLinesUtf8() //
23                  .comparator((a, b) -> {
24                      String ida = a.substring(0, a.indexOf(' '));
25                      String idb = b.substring(0, b.indexOf(' '));
26                      return ida.compareTo(idb);
27                  }) //
28                  .input(input) //
29                  .output(output) //
30                  .sort();
31  
32          Files.readAllLines(output.toPath()).forEach(System.out::println);
33      }
34  
35  }
