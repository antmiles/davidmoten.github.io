21       * the close the coverage is to the region in question.

23      private final double ratio;

33      public CoverageLongs(long[] hashes, int count, double ratio) {

43       * @return set of hashes
44       */
45      public long[] getHashes() {
46          long[] res = new long[count];
47          System.arraycopy(hashes, 0, res, 0, count);
48          return res;
49      }
50  
51      /**
52       * Returns the measure of how well the hashes cover a region. The ratio is
53       * the total area of hashes divided by the area of the bounding box in
54       * degrees squared. The closer the ratio is to 1 the better the more closely
55       * the hashes approximate the bounding box.
56       * 
57       * @return ratio of area of hashes to area of target region.
58       */
59      public double getRatio() {
60          return ratio;
61      }
62  
63      /**
64       * Returns the length in characters of the first hash returned by an
65       * iterator on the hash set. All hashes should be of the same length in this
66       * coverage.
67       * 
68       * @return length of the hash
69       */
70      public int getHashLength() {
71          if (count == 0)
72              return 0;
73          else
74              return (int) (hashes[0] & 0x0f);
75      }
76

80      }
