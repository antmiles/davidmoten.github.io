128     /**
129      * Returns the adjacent hash in given {@link Direction}. Based on
130      * https://github.com/davetroy/geohash-js/blob/master/geohash.js. This
131      * method is an improvement on the original js method because it works at
132      * borders too (at the poles and the -180,180 longitude boundaries).
133      * 
134      * @param hash
135      *            string hash relative to which the adjacent is returned
136      * @param direction
137      *            direction relative to {@code hash} for which the adjacent is
138      *            returned
139      * @return hash of adjacent hash
140      */
141     public static String adjacentHash(String hash, Direction direction) {
142         checkHash(hash);
143         Preconditions.checkArgument(hash.length() > 0,
144                 "adjacent has no meaning for a zero length hash that covers the whole world");
145 
146         String adjacentHashAtBorder = adjacentHashAtBorder(hash, direction);
147         if (adjacentHashAtBorder != null)
148             return adjacentHashAtBorder;
149 
150         String source = hash.toLowerCase();
151         char lastChar = source.charAt(source.length() - 1);
152         Parity parity = (source.length() % 2 == 0) ? Parity.EVEN : Parity.ODD;
153         String base = source.substring(0, source.length() - 1);
154         if (BORDERS.get(direction).get(parity).indexOf(lastChar) != -1)
155             base = adjacentHash(base, direction);
156         return base + BASE32.charAt(NEIGHBOURS.get(direction).get(parity).indexOf(lastChar));
157     }

159     private static String adjacentHashAtBorder(String hash, Direction direction) {
160         // check if hash is on edge and direction would push us over the edge
161         // if so, wrap round to the other limit for longitude
162         // or if at latitude boundary (a pole) then spin longitude around 180
163         // degrees.
164         LatLong centre = decodeHash(hash);
165 
166         // if rightmost hash
167         if (Direction.RIGHT.equals(direction)) {
168             if (Math.abs(centre.getLon() + widthDegrees(hash.length()) / 2 - 180) < PRECISION) {
169                 return encodeHash(centre.getLat(), -180, hash.length());
170             }
171         }
172         // if leftmost hash
173         else if (Direction.LEFT.equals(direction)) {
174             if (Math.abs(centre.getLon() - widthDegrees(hash.length()) / 2 + 180) < PRECISION) {
175                 return encodeHash(centre.getLat(), 180, hash.length());
176             }
177         }
178         // if topmost hash
179         else if (Direction.TOP.equals(direction)) {
180             if (Math.abs(centre.getLat() + widthDegrees(hash.length()) / 2 - 90) < PRECISION) {
181                 return encodeHash(centre.getLat(), centre.getLon() + 180, hash.length());
182             }
183         }
184         // if bottommost hash
185         else {
186             if (Math.abs(centre.getLat() - widthDegrees(hash.length()) / 2 + 90) < PRECISION) {
187                 return encodeHash(centre.getLat(), centre.getLon() + 180, hash.length());
188             }
189         }
190 
191         return null;
192     }

250      * Returns the adjacent hash N steps in the given {@link Direction}. A
251      * negative N will use the opposite {@link Direction}.
252      * 
253      * @param hash
254      *            origin hash
255      * @param direction
256      *            to desired hash
257      * @param steps
258      *            number of hashes distance to desired hash
259      * @return hash at position in direction a number of hashes away (steps)
260      */
261     public static String adjacentHash(String hash, Direction direction, int steps) {
262         if (steps < 0)
263             return adjacentHash(hash, direction.opposite(), Math.abs(steps));
264         else {
265             String h = hash;
266             for (int i = 0; i < steps; i++)
267                 h = adjacentHash(h, direction);
268             return h;
269         }
270     }

282         String left = adjacentHash(hash, Direction.LEFT);
283         String right = adjacentHash(hash, Direction.RIGHT);
284         list.add(left);
285         list.add(right);
286         list.add(adjacentHash(hash, Direction.TOP));
287         list.add(adjacentHash(hash, Direction.BOTTOM));
288         list.add(adjacentHash(left, Direction.TOP));
289         list.add(adjacentHash(left, Direction.BOTTOM));
290         list.add(adjacentHash(right, Direction.TOP));
291         list.add(adjacentHash(right, Direction.BOTTOM));
292         return list;

347     // Translated to java from:
348     // geohash.js
349     // Geohash library for Javascript

352     public static String encodeHash(double latitude, double longitude, int length) {
353         Preconditions.checkArgument(length > 0, "length must be greater than zero");
354         Preconditions.checkArgument(latitude >= -90 && latitude <= 90,
355                 "latitude must be between -90 and 90 inclusive");
356         longitude = Position.to180(longitude);
357 
358         return fromLongToString(encodeHashToLong(latitude, longitude, length));
359     }

361     /**
362      * Takes a hash represented as a long and returns it as a string.
363      *
364      * @param hash
365      *            the hash, with the length encoded in the 4 least significant
366      *            bits
367      * @return the string encoded geohash
368      */
369     static String fromLongToString(long hash) {
370         int length = (int) (hash & 0xf);
371         if (length > 12 || length < 1)
372             throw new IllegalArgumentException("invalid long geohash " + hash);
373         char[] geohash = new char[length];
374         for (int pos = 0; pos < length; pos++) {
375             geohash[pos] = BASE32.charAt(((int) (hash >>> 59)));
376             hash <<= 5;
377         }
378         return new String(geohash);
379     }

381     static long encodeHashToLong(double latitude, double longitude, int length) {
382         boolean isEven = true;
383         double minLat = -90.0, maxLat = 90;
384         double minLon = -180.0, maxLon = 180.0;
385         long bit = 0x8000000000000000L;
386         long g = 0;
387 
388         long target = 0x8000000000000000L >>> (5 * length);
389         while (bit != target) {
390             if (isEven) {
391                 double mid = (minLon + maxLon) / 2;
392                 if (longitude >= mid) {
393                     g |= bit;
394                     minLon = mid;
395                 } else
396                     maxLon = mid;
397             } else {
398                 double mid = (minLat + maxLat) / 2;
399                 if (latitude >= mid) {
400                     g |= bit;
401                     minLat = mid;
402                 } else
403                     maxLat = mid;
404             }
405 
406             isEven = !isEven;
407             bit >>>= 1;
408         }
409         return g |= length;
410     }

412     /**
413      * Returns a latitude,longitude pair as the centre of the given geohash.
414      * Latitude will be between -90 and 90 and longitude between -180 and 180.
415      * 
416      * @param geohash
417      *            hash to decode
418      * @return lat long point
419      */
420     // Translated to java from:
421     // geohash.js
422     // Geohash library for Javascript
423     // (c) 2008 David Troy
424     // Distributed under the MIT License
425     public static LatLong decodeHash(String geohash) {
426         Preconditions.checkNotNull(geohash, "geohash cannot be null");
427         boolean isEven = true;
428         double[] lat = new double[2];
429         double[] lon = new double[2];
430         lat[0] = -90.0;
431         lat[1] = 90.0;
432         lon[0] = -180.0;
433         lon[1] = 180.0;
434 
435         for (int i = 0; i < geohash.length(); i++) {
436             char c = geohash.charAt(i);
437             int cd = BASE32.indexOf(c);
438             for (int j = 0; j < 5; j++) {
439                 int mask = BITS[j];
440                 if (isEven) {
441                     refineInterval(lon, cd, mask);
442                 } else {
443                     refineInterval(lat, cd, mask);
444                 }
445                 isEven = !isEven;
446             }
447         }
448         double resultLat = (lat[0] + lat[1]) / 2;
449         double resultLon = (lon[0] + lon[1]) / 2;
450 
451         return new LatLong(resultLat, resultLon);
452     }

454     /**
455      * Refines interval by a factor or 2 in either the 0 or 1 ordinate.
456      * 
457      * @param interval
458      *            two entry array of double values
459      * @param cd
460      *            used with mask
461      * @param mask
462      *            used with cd
463      */
464     private static void refineInterval(double[] interval, int cd, int mask) {
465         if ((cd & mask) != 0)
466             interval[0] = (interval[0] + interval[1]) / 2;
467         else
468             interval[1] = (interval[0] + interval[1]) / 2;
469     }

471     /**
472      * Returns the maximum length of hash that covers the bounding box. If no
473      * hash can enclose the bounding box then 0 is returned.
474      * 
475      * @param topLeftLat
476      *            latitude of top left point (north west)
477      * @param topLeftLon
478      *            longitude of top left point (north west)
479      * @param bottomRightLat
480      *            latitude of bottom right point (south east)
481      * @param bottomRightLon
482      *            longitude of bottom right point (south east)
483      * @return length of the hash
484      */
485     public static int hashLengthToCoverBoundingBox(double topLeftLat, double topLeftLon,
486             double bottomRightLat, double bottomRightLon) {
487         boolean isEven = true;
488         double minLat = -90.0, maxLat = 90;
489         double minLon = -180.0, maxLon = 180.0;
490 
491         for (int bits = 0; bits < MAX_HASH_LENGTH * 5; bits++) {
492             if (isEven) {
493                 double mid = (minLon + maxLon) / 2;
494                 if (topLeftLon >= mid) {
495                     if (bottomRightLon < mid)
496                         return bits / 5;
497                     minLon = mid;
498                 } else {
499                     if (bottomRightLon >= mid)
500                         return bits / 5;
501                     maxLon = mid;
502                 }
503             } else {
504                 double mid = (minLat + maxLat) / 2;
505                 if (topLeftLat >= mid) {
506                     if (bottomRightLat < mid)
507                         return bits / 5;
508                     minLat = mid;
509                 } else {
510                     if (bottomRightLat >= mid)
511                         return bits / 5;
512                     maxLat = mid;
513                 }
514             }
515 
516             isEven = !isEven;
517         }
518         return MAX_HASH_LENGTH;
519     }

533     public static boolean hashContains(String hash, double lat, double lon) {
534         LatLong centre = decodeHash(hash);
535         return Math.abs(centre.getLat() - lat) <= heightDegrees(hash.length()) / 2
536                 && Math.abs(to180(centre.getLon() - lon)) <= widthDegrees(hash.length()) / 2;

560     /**
561      * Returns the hashes that are required to cover the given bounding box. The
562      * maximum length of hash is selected that satisfies the number of hashes
563      * returned is less than <code>maxHashes</code>. Returns null if hashes
564      * cannot be found satisfying that condition. Maximum hash length returned
565      * will be {@link GeoHash}.MAX_HASH_LENGTH.
566      * 
567      * @param topLeftLat
568      *            latitude of top left point (north west)
569      * @param topLeftLon
570      *            longitude of top left point (north west)
571      * @param bottomRightLat
572      *            latitude of bottom right point (south east)
573      * @param bottomRightLon
574      *            longitude of bottom right point (south east)
575      * @param maxHashes
576      *            maximum number of hashes to use to cover the box
577      * @return coverage
578      */
579     public static Coverage coverBoundingBoxMaxHashes(double topLeftLat, final double topLeftLon,
580             final double bottomRightLat, final double bottomRightLon, int maxHashes) {
581         CoverageLongs coverage = null;
582         int startLength = hashLengthToCoverBoundingBox(topLeftLat, topLeftLon, bottomRightLat,
583                 bottomRightLon);
584         if (startLength == 0)
585             startLength = 1;
586         for (int length = startLength; length <= MAX_HASH_LENGTH; length++) {
587             CoverageLongs c = coverBoundingBoxLongs(topLeftLat, topLeftLon, bottomRightLat,
588                     bottomRightLon, length);
589             if (c.getCount() > maxHashes)
590                 return coverage == null ? null : new Coverage(coverage);
591             else
592                 coverage = c;
593         }
594         // note coverage can never be null
595         return new Coverage(coverage);
596     }

621     private static class LongSet {
622         int count = 0;
623         private int cap = 16;
624         long[] array = new long[cap];
625 
626         void add(long l) {
627             for (int i = 0; i < count; i++)
628                 if (array[i] == l)
629                     return;
630             if (count == cap) {
631                 long[] newArray = new long[cap *= 2];
632                 System.arraycopy(array, 0, newArray, 0, count);
633                 array = newArray;
634             }
635             array[count++] = l;
636         }
637     }

639     static CoverageLongs coverBoundingBoxLongs(double topLeftLat, final double topLeftLon,
640             final double bottomRightLat, final double bottomRightLon, final int length) {
641         Preconditions.checkArgument(topLeftLat >= bottomRightLat,
642                 "topLeftLat must be >= bottomRighLat");
643         Preconditions.checkArgument(topLeftLon <= bottomRightLon,
644                 "topLeftLon must be <= bottomRighLon");
645         Preconditions.checkArgument(length > 0, "length must be greater than zero");
646         final double actualWidthDegreesPerHash = widthDegrees(length);
647         final double actualHeightDegreesPerHash = heightDegrees(length);
648 
649         LongSet hashes = new LongSet();
650 
651         double diff = Position.longitudeDiff(bottomRightLon, topLeftLon);
652         double maxLon = topLeftLon + diff;
653 
654         for (double lat = bottomRightLat; lat <= topLeftLat; lat += actualHeightDegreesPerHash) {
655             for (double lon = topLeftLon; lon <= maxLon; lon += actualWidthDegreesPerHash) {
656                 hashes.add(encodeHashToLong(lat, lon, length));
657             }
658         }
659         // ensure have the borders covered
660         for (double lat = bottomRightLat; lat <= topLeftLat; lat += actualHeightDegreesPerHash) {
661             hashes.add(encodeHashToLong(lat, maxLon, length));
662         }
663         for (double lon = topLeftLon; lon <= maxLon; lon += actualWidthDegreesPerHash) {
664             hashes.add(encodeHashToLong(topLeftLat, lon, length));
665         }
666         // ensure that the topRight corner is covered
667         hashes.add(encodeHashToLong(topLeftLat, maxLon, length));
668 
669         double areaDegrees = diff * (topLeftLat - bottomRightLat);
670         double coverageAreaDegrees = hashes.count * widthDegrees(length) * heightDegrees(length);
671         double ratio = coverageAreaDegrees / areaDegrees;

823      * 
824      * @param hash reference hash
825      * @param fromRight
826      *            top left of the grid in hashes to the right (can be negative).
827      * @param fromBottom
828      *            top left of the grid in hashes to the bottom (can be
829      *            negative).
830      * @param toRight
831      *            bottom righth of the grid in hashes to the bottom (can be
832      *            negative).
833      * @param toBottom
834      *            bottom right of the grid in hashes to the bottom (can be
835      *            negative).
836      * @param highlightThese hashes to highlight
837      * @return String representation of grid
838      */
839     public static String gridAsString(String hash, int fromRight, int fromBottom, int toRight,
840             int toBottom, Set<String> highlightThese) {
841         StringBuilder s = new StringBuilder();
842         for (int bottom = fromBottom; bottom <= toBottom; bottom++) {
843             for (int right = fromRight; right <= toRight; right++) {
844                 String h = adjacentHash(hash, Direction.RIGHT, right);
845                 h = adjacentHash(h, Direction.BOTTOM, bottom);
846                 if (highlightThese.contains(h))
847                     h = h.toUpperCase();
848                 s.append(h).append(" ");
849             }

851         }
