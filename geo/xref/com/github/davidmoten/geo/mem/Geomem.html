19  /**
20   * Provides fast concurrent querying using in memory
21   * {@link ConcurrentSkipListMap}s and geohash to store data with time and
22   * position. Depends on guava library.
23   * 
24   * @author dxm
25   * 
26   * @param <T>
27   *            The type of the record with position and time.
28   * @param <R>
29   *            The type of the id of the record with position and time.
30   */
31  public class Geomem<T, R> {
32  
33      /**
34       * Maps from base32 geohash (long) to a map of time in epoch ms to
35       * {@link Info}.
36       */
37      private final Map<Long, SortedMap<Long, Info<T, R>>> mapByGeoHash = Maps
38              .newConcurrentMap();
39  
40      /**
41       * Records a mapByGeoHash as above for each id of type R.
42       */
43      private final Map<R, Map<Long, SortedMap<Long, Info<T, R>>>> mapById = Maps
44              .newConcurrentMap();
45  
46      /**
47       * Returns as an {@link Iterable} the results of a search within the
48       * bounding box given and where start &lt;=time &lt; finish.
49       * 
50       * @param topLeftLat
51       * @param topLeftLong
52       * @param bottomRightLat
53       * @param bottomRightLong
54       * @param start
55       * @param finish
56       * @return
57       */
58      public Iterable<Info<T, R>> find(double topLeftLat, double topLeftLong,
59              double bottomRightLat, double bottomRightLong, long start,
60              long finish) {
61  
62          Coverage cover = GeoHash.coverBoundingBox(topLeftLat, topLeftLong,
63                  bottomRightLat, bottomRightLong);
64          Iterable<Info<T, R>> it = Collections.emptyList();
65          for (String hash : cover.getHashes()) {
66              it = Iterables.concat(
67                      it,
68                      find(topLeftLat, topLeftLong, bottomRightLat,
69                              bottomRightLong, start, finish, hash));
70          }
71          return it;
72      }

74      /**
75       * Returns an {@link Iterable} of {@link Info} being those records within
76       * the bounding box, start<=time < finish and inside the geoHash withinHash.
77       * 
78       * Filters first on withinHash, then time, then bounding box.
79       * 
80       * @param topLeftLat
81       * @param topLeftLong
82       * @param bottomRightLat
83       * @param bottomRightLong
84       * @param start
85       * @param finish
86       * @param withinHash
87       * @return
88       */
89      private Iterable<Info<T, R>> find(final double topLeftLat,
90              final double topLeftLong, final double bottomRightLat,
91              final double bottomRightLong, long start, long finish,
92              String withinHash) {
93  
94          Iterable<Info<T, R>> it = find(start, finish, withinHash);
95          return Iterables.filter(
96                  it,
97                  createRegionFilter(topLeftLat, topLeftLong, bottomRightLat,
98                          bottomRightLong));
99      }

101     /**
102      * Returns a {@link Predicate} that returns true if and only if a point is
103      * within the bounding box, exclusive of the top (north) and left (west)
104      * edges.
105      * 
106      * @param topLeftLat
107      * @param topLeftLong
108      * @param bottomRightLat
109      * @param bottomRightLong
110      * @return
111      */
112     @VisibleForTesting
113     Predicate<Info<T, R>> createRegionFilter(final double topLeftLat,
114             final double topLeftLong, final double bottomRightLat,
115             final double bottomRightLong) {
116         return new Predicate<Info<T, R>>() {
117 
118             @Override
119             public boolean apply(Info<T, R> info) {
120                 return info.lat() >= bottomRightLat && info.lat() < topLeftLat
121                         && info.lon() > topLeftLong
122                         && info.lon() <= bottomRightLong;
123             }
124         };
125     }

127     /**
128      * Returns the {@link Info}s where start<=time <finish and position is
129      * inside the geohash withinHash.
130      * 
131      * @param start
132      * @param finish
133      * @param withinHash
134      * @return
135      */
136     private Iterable<Info<T, R>> find(long start, long finish, String withinHash) {
137         long key = Base32.decodeBase32(withinHash);
138         SortedMap<Long, Info<T, R>> sortedByTime = mapByGeoHash.get(key);
139         if (sortedByTime == null)
140             return Collections.emptyList();
141         else
142             return sortedByTime.subMap(start, finish).values();
143     }

188     /**
189      * Adds a record to the in-memory store with the given position, time and
190      * id.
191      * 
192      * @param info
193      */
194     public void add(Info<T, R> info) {
195         String hash = GeoHash.encodeHash(info.lat(), info.lon());
196 
197         addToMap(mapByGeoHash, info, hash);
198         addToMapById(mapById, info, hash);
199     }

201     /**
202      * Adds {@link Info}to the map by id.
203      * 
204      * @param mapById
205      * @param info
206      * @param hash
207      */
208     private void addToMapById(
209             Map<R, Map<Long, SortedMap<Long, Info<T, R>>>> mapById,
210             Info<T, R> info, String hash) {
211         if (info.id().isPresent()) {
212             Map<Long, SortedMap<Long, Info<T, R>>> m = mapById.get(info.id()
213                     .get());
214             synchronized (mapByGeoHash) {
215                 if (m == null) {
216                     m = Maps.newConcurrentMap();
217                     mapById.put(info.id().get(), m);
218                 }
219             }
220             addToMap(m, info, hash);
221         }
222     }

224     /**
225      * Adds {@link Info} to the map by geohash.
226      * 
227      * @param map
228      * @param info
229      * @param hash
230      */
231     private void addToMap(Map<Long, SortedMap<Long, Info<T, R>>> map,
232             Info<T, R> info, String hash) {
233 
234         // full hash length is 12 so this will insert 12 entries
235         for (int i = 1; i <= hash.length(); i++) {
236             long key = Base32.decodeBase32(hash.substring(0, i));
237             synchronized (map) {
238                 if (map.get(key) == null) {
239                     map.put(key, new ConcurrentSkipListMap<Long, Info<T, R>>());
240                 }
241             }
242             map.get(key).put(info.time(), info);
243         }
244     }
