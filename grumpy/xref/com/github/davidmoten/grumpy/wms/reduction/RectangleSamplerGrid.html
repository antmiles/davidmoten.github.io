9   /**
10   * Samples bounds by creating a grid starting at top left corner of
11   * min(maxSizeKm,widthKm) by min(maxSizeKm, heightKm). Always includes bottom
12   * and right edge points as well.
13   */
14  public class RectangleSamplerGrid implements RectangleSampler {
15  
16      private final double maxCellSizeKm;
17  
18      public RectangleSamplerGrid(double maxCellSizeKm) {
19          this.maxCellSizeKm = maxCellSizeKm;
20      }
21  
22      @Override
23      public List<Point> sample(Rectangle region, Projector projector) {
24  
25          return RectangleUtil.corners(region);
26      }
27  }
