12      @Test
13      public void testDoOnError() {
14          RuntimeException error = new RuntimeException("boo");
15          AtomicBoolean b = new AtomicBoolean();
16          Single //
17                  .error(error) //
18                  .doOnError(e -> b.set(e == error)) //
19                  .test() //
20                  .assertError(RuntimeException.class);
21          assertTrue(b.get());
22      }
