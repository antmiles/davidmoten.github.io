5   public final class MaybeTester<T> {
6   
7       private final Optional<T> value;
8   
9       public MaybeTester(Maybe<T> maybe) {
10          value = maybe.get();
11      }
12  
13      public void assertValue(T t) {
14          if (!value.isPresent()) {
15              throw new AssertionError("Value " + t + " expected but no value found");
16          } else if (!value.get().equals(t)) {
17              throw new AssertionError("Value " + t + " expected but found " + value.get());
18          }
19      }
20  
21      public void assertNoValue() {
22          if (value.isPresent()) {
23              throw new AssertionError("Did not expect a value but found " + value.get());
24          }
25      }
26  
27  }
