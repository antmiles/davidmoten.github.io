3   import java.util.NoSuchElementException;
4   import java.util.function.BiFunction;
5   import java.util.function.Supplier;
6   
7   import org.davidmoten.kool.Stream;
8   import org.davidmoten.kool.StreamIterable;
9   import org.davidmoten.kool.StreamIterator;
10  
11  import com.github.davidmoten.guavamini.Preconditions;
12  
13  public final class ReduceWithInitialValueSupplier<R, T> implements Stream<R> {
14  
15      private final Supplier<R> initialValue;
16      private final BiFunction<? super R, ? super T, ? extends R> reducer;
17      private final StreamIterable<T> source;
18  
19      public ReduceWithInitialValueSupplier(Supplier<R> initialValue,
20              BiFunction<? super R, ? super T, ? extends R> reducer, StreamIterable<T> source) {
21          this.initialValue = initialValue;
22          this.reducer = reducer;
23          this.source = source;
24      }
25  
26      @Override
27      public StreamIterator<R> iterator() {
28          return new StreamIterator<R>() {
29  
30              final StreamIterator<T> it = Preconditions.checkNotNull(source.iterator());
31              R value;
32              boolean finished;
33  
34              @Override
35              public boolean hasNext() {
36                  if (finished) {
37                      return false;
38                  } else {
39                      calculate();
40                      return true;
41                  }
42              }
43  
44              @Override
45              public R next() {
46                  if (finished) {
47                      throw new NoSuchElementException();
48                  } else {
49                      finished = true;
50                      calculate();
51                      R r = value;
52                      value = null;
53                      return r;
54                  }
55              }
56  
57              private void calculate() {
58                  R r = Preconditions.checkNotNull(initialValue.get());
59                  while (it.hasNext()) {
60                      r = Preconditions.checkNotNull(reducer.apply(r, it.next()));
61                  }
62                  value = r;
63              }
64  
65              @Override
66              public void dispose() {
67                  it.dispose();
68              }
69  
70          };
71      }
72  
73  }
