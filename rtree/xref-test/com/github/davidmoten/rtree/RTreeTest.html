49      @Test
50      public void testPerformanceAndEntriesCount() {
51  
52          long repeats = Long.parseLong(System.getProperty("r", "1"));
53          long n = Long.parseLong(System.getProperty("n", "10000"));
54          RTree<Object> tree = null;
55          while (--repeats >= 0) {
56              long t = System.currentTimeMillis();
57              tree = createRandomRTree(n);
58              long diff = System.currentTimeMillis() - t;
59              System.out.println("inserts/second = " + ((double) n / diff * 1000));
60          }
61          assertEquals(n, (int) tree.entries().count().toBlocking().single());
62  
63          long t = System.currentTimeMillis();
64          Entry<Object> entry = tree.search(rectangle(0, 0, 500, 500)).first().toBlocking().single();
65          long diff = System.currentTimeMillis() - t;
66          System.out.println("found " + entry);
67          System.out.println("time to get nearest with " + n + " entries=" + diff);
68  
69      }

71      private static RTree<Object> createRandomRTree(long n) {
72          RTree<Object> tree = RTree.maxChildren(4).create();
73          for (long i = 0; i < n; i++) {
74              Entry<Object> entry = entry(new Object(), random());
75              tree = tree.add(entry);
76          }
77          return tree;
78      }

125     public void testDeletionThatRemovesAllNodesChildren() {
126         RTree<Object> tree = create(3, 8);
127         tree = tree.add(e(10));
128         // node children are now 1,2 and 3,4
129         assertEquals(3, tree.calculateDepth());
130         tree = tree.delete(e(10));
131         // node children are now 1,2 and 3
132         assertEquals(3, tree.calculateDepth());
133         assertEquals(Sets.newHashSet(e(1), e(2), e(3), e(4), e(5), e(6), e(7), e(8)),

193     private static RTree<Object> create(int maxChildren, int n) {
194         RTree<Object> tree = RTree.maxChildren(maxChildren).create();
195         for (int i = 1; i <= n; i++)
196             tree = tree.add(e(i));
197         return tree;
198     }

204         List<Entry<Object>> list = tree.nearest(r(9), 10, 2).toList().toBlocking().single();

234     public void testDeleteOneFromLargeTree() {
235         int n = 10000;
236         RTree<Object> tree = createRandomRTree(n).add(e(1)).add(e(2)).delete(e(1));
237         assertEquals(n + 1, (int) tree.entries().count().toBlocking().single());
238         assertFalse(tree.entries().contains(e(1)).toBlocking().single());
239         assertTrue(tree.entries().contains(e(2)).toBlocking().single());
240         for (Entry<Object> entry : tree.entries().toBlocking().toIterable()) {
241             tree = tree.delete(entry);
242         }
243         assertEquals(0, (int) tree.entries().count().toBlocking().single());
244         assertTrue(tree.isEmpty());
