49      private void roundTrip(InternalStructure structure, boolean backpressure) throws Exception {
50          RTree<Object, Point> tree = RTree.star().maxChildren(10).create();
51          tree = tree.add(GreekEarthquakes.entries()).last().toBlocking().single();
52          long t = System.currentTimeMillis();
53          File file = new File("target/file");
54          FileOutputStream os = new FileOutputStream(file);
55          Serializer<Object, Point> fbSerializer = createSerializer();
56  
57          serialize(tree, t, file, os, fbSerializer);
58  
59          deserialize(structure, file, fbSerializer, backpressure);
60      }
61  
62      private static Serializer<Object, Point> createSerializer() {
63          Func1<Object, byte[]> serializer = new Func1<Object, byte[]>() {
64              @Override
65              public byte[] call(Object o) {
66                  return EMPTY;
67              }
68          };
69          Func1<byte[], Object> deserializer = new Func1<byte[], Object>() {
70              @Override
71              public Object call(byte[] bytes) {
72                  return null;
73              }
74          };
75          Serializer<Object, Point> fbSerializer = SerializerFlatBuffers.create(serializer,
76                  deserializer);
77          return fbSerializer;
78      }
79  
80      private static void serialize(RTree<Object, Point> tree, long t, File file, FileOutputStream os,
81              Serializer<Object, Point> fbSerializer) throws IOException {
82          fbSerializer.write(tree, os);
83          os.close();
84          System.out.println("written in " + (System.currentTimeMillis() - t) + "ms, " + "file size="
85                  + file.length() / 1000000.0 + "MB");
86          System.out.println("bytes per entry=" + file.length() / tree.size());
87      }
88  
89      private static void deserialize(InternalStructure structure, File file,
90              Serializer<Object, Point> fbSerializer, boolean backpressure) throws Exception {
91          long t = System.currentTimeMillis();
92          InputStream is = new FileInputStream(file);
93          t = System.currentTimeMillis();
94          RTree<Object, Point> tr = fbSerializer.read(is, file.length(), structure);
95          System.out.println(tr.root().get());
96  
97          System.out.println("read in " + (System.currentTimeMillis() - t) + "ms");
98          Observable<Entry<Object, Point>> o = tr.search(Geometries.rectangle(40, 27.0, 40.5, 27.5));
99          if (backpressure)
100             o = o.take(10000);
101         int found = o.count().toBlocking().single();
102         System.out.println("found=" + found);
103         assertEquals(22, found);
104         System.out.println(tr.size());
105     }
