1   package com.github.davidmoten.rtree;

3   import rx.Subscriber;
4   import rx.functions.Func1;
5   
6   import com.github.davidmoten.rtree.geometry.Geometry;
7   import com.github.davidmoten.util.ImmutableStack;
8   
9   /**
10   * Utility methods for controlling backpressure of the tree search.
11   */
12  final class Backpressure {
13  
14      private Backpressure() {
15          // prevent instantiation
16      }
17  
18      static <T, S extends Geometry> ImmutableStack<NodePosition<T, S>> search(
19              final Func1<? super Geometry, Boolean> condition,
20              final Subscriber<? super Entry<T, S>> subscriber,
21              ImmutableStack<NodePosition<T, S>> stack, long request) {
22          if (stack.isEmpty())
23              return stack;
24          while (true) {
25              NodePosition<T, S> np = stack.peek();
26              if (subscriber.isUnsubscribed())
27                  return ImmutableStack.empty();
28              else if (request == 0)
29                  return stack;
30              else if (np.position() == np.node().count()) {
31                  // handle after last position in node
32                  ImmutableStack<NodePosition<T, S>> stack2 = stack.pop();
33                  if (stack2.isEmpty())
34                      return stack2;
35                  else {
36                      NodePosition<T, S> previous = stack2.peek();
37                      stack = stack2.pop().push(previous.nextPosition());
38                  }
39              } else if (np.node() instanceof NonLeaf) {
40                  // handle non-leaf
41                  Node<T, S> child = ((NonLeaf<T, S>) np.node()).children().get(np.position());
42                  if (condition.call(child.geometry())) {
43                      stack = stack.push(new NodePosition<T, S>(child, 0));
44                  } else {
45                      stack = stack.pop().push(np.nextPosition());
46                  }
47              } else {
48                  // handle leaf
49                  Entry<T, S> entry = ((Leaf<T, S>) np.node()).entries().get(np.position());
50                  final long nextRequest;
51                  if (condition.call(entry.geometry())) {
52                      subscriber.onNext(entry);
53                      nextRequest = request - 1;
54                  } else
55                      nextRequest = request;
56                  stack = stack.pop().push(np.nextPosition());
57                  request = nextRequest;
58              }
59          }
60      }
61  
62  }
