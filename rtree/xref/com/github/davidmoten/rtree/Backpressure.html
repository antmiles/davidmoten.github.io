1   package com.github.davidmoten.rtree;
2   
3   import rx.Subscriber;
4   import rx.functions.Func1;
5   
6   import com.github.davidmoten.rtree.geometry.Geometry;
7   import com.github.davidmoten.util.ImmutableStack;
8   
9   /**
10   * Utility methods for controlling backpressure of the tree search.
11   */
12  final class Backpressure {
13  
14      private Backpressure() {
15          // prevent instantiation
16      }
17  
18      static <T, S extends Geometry> ImmutableStack<NodePosition<T, S>> search(
19              final Func1<? super Geometry, Boolean> condition,
20              final Subscriber<? super Entry<T, S>> subscriber,
21              ImmutableStack<NodePosition<T, S>> stack, long request) {
22          while (!stack.isEmpty()) {
23              NodePosition<T, S> np = stack.peek();
24              if (subscriber.isUnsubscribed())
25                  return ImmutableStack.empty();
26              else if (request == 0)
27                  return stack;
28              else if (np.position() == np.node().count()) {
29                  // handle after last in node
30                  stack = searchAfterLastInNode(stack);
31              } else if (np.node() instanceof NonLeaf) {
32                  // handle non-leaf
33                  stack = searchNonLeaf(condition, stack, np);
34              } else {
35                  // handle leaf
36                  long nextRequest = searchLeaf(condition, subscriber, request, np);
37                  stack = stack.pop().push(np.nextPosition());
38                  request = nextRequest;
39              }
40          }
41          return stack;
42      }
43  
44      private static <T, S extends Geometry> long searchLeaf(
45              final Func1<? super Geometry, Boolean> condition,
46              final Subscriber<? super Entry<T, S>> subscriber, long request, NodePosition<T, S> np) {
47          final long nextRequest;
48          Entry<T, S> entry = ((Leaf<T, S>) np.node()).entries().get(np.position());
49          if (condition.call(entry.geometry())) {
50              subscriber.onNext(entry);
51              nextRequest = request - 1;
52          } else
53              nextRequest = request;
54          return nextRequest;
55      }
56  
57      private static <S extends Geometry, T> ImmutableStack<NodePosition<T, S>> searchAfterLastInNode(
58              ImmutableStack<NodePosition<T, S>> stack) {
59          ImmutableStack<NodePosition<T, S>> stack2 = stack.pop();
60          if (stack2.isEmpty())
61              stack = stack2;
62          else {
63              NodePosition<T, S> previous = stack2.peek();
64              stack = stack2.pop().push(previous.nextPosition());
65          }
66          return stack;
67      }
68  
69      private static <S extends Geometry, T> ImmutableStack<NodePosition<T, S>> searchNonLeaf(
70              final Func1<? super Geometry, Boolean> condition,
71              ImmutableStack<NodePosition<T, S>> stack, NodePosition<T, S> np) {
72          Node<T, S> child = ((NonLeaf<T, S>) np.node()).children().get(np.position());
73          if (condition.call(child.geometry())) {
74              stack = stack.push(new NodePosition<T, S>(child, 0));
75          } else {
76              stack = stack.pop().push(np.nextPosition());
77          }
78          return stack;
79      }

81  }
