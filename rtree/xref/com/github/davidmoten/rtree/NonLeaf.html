6   import rx.Subscriber;

9   import com.github.davidmoten.util.ImmutableStack;
10  import com.google.common.base.Optional;
11  import com.google.common.base.Preconditions;

13  final class NonLeaf implements Node {
14  
15  	private final List<? extends Node> children;
16  	private final Rectangle mbr;
17  
18  	NonLeaf(List<? extends Node> children) {
19  		Preconditions.checkArgument(!children.isEmpty());
20  		this.children = children;
21  		this.mbr = Util.mbr(children);
22  	}
23  
24  	List<? extends Node> children() {
25  		return children;
26  	}
27  
28  	@Override
29  	public Rectangle mbr() {
30  		return mbr;
31  	}
32  
33  	@Override
34  	public Node add(Entry entry, ImmutableStack<NonLeaf> stack) {
35  		final HasMbr child = Util.findLeastIncreaseInMbrArea(entry.mbr(),
36  				children);
37  		return ((Node) child).add(entry, stack.push(this));
38  	}
39  
40  	@Override
41  	public void search(Func1<? super Rectangle, Boolean> criterion,
42  			Subscriber<? super Entry> subscriber,
43  			Optional<Comparator<Rectangle>> comparator) {
44  
45  		for (Node child : Util.sort(children, comparator)) {
46  			if (subscriber.isUnsubscribed())
47  				return;
48  			else {
49  				if (criterion.call(child.mbr()))
50  					child.search(criterion, subscriber, comparator);
51  			}
52  		}
53  	}
54  
55  	@Override
56  	public String toString() {
57  		return "NonLeaf [mbr=" + mbr + "]";
58  	}
59  
60  }
