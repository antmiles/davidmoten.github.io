3   import static com.google.common.base.Optional.of;
4   
5   import java.util.ArrayList;
6   import java.util.Collections;
7   import java.util.List;
8   
9   import rx.Subscriber;
10  import rx.functions.Func1;
11  
12  import com.github.davidmoten.rtree.geometry.Geometry;
13  import com.github.davidmoten.rtree.geometry.ListPair;
14  import com.github.davidmoten.rtree.geometry.Rectangle;
15  import com.google.common.base.Optional;
16  import com.google.common.base.Preconditions;
17  
18  final class NonLeaf<T, S extends Geometry> implements Node<T, S> {
19  
20      private final List<? extends Node<T, S>> children;
21      private final Rectangle mbr;
22      private final Context context;
23  
24      NonLeaf(List<? extends Node<T, S>> children, Context context) {
25          Preconditions.checkArgument(!children.isEmpty());
26          this.context = context;
27          this.children = children;
28          this.mbr = Util.mbr(children);
29      }
30  
31      @Override
32      public Geometry geometry() {
33          return mbr;
34      }
35  
36      @Override
37      public void search(Func1<? super Geometry, Boolean> criterion,
38              Subscriber<? super Entry<T, S>> subscriber) {
39  
40          if (!criterion.call(this.geometry().mbr()))
41              return;
42  
43          for (final Node<T, S> child : children) {
44              if (subscriber.isUnsubscribed())
45                  return;
46              else
47                  child.search(criterion, subscriber);
48          }
49      }
50  
51      @Override
52      public int count() {
53          return children.size();
54      }
55  
56      List<? extends Node<T, S>> children() {
57          return children;
58      }
59  
60      @Override
61      public List<Node<T, S>> add(Entry<? extends T, ? extends S> entry) {
62          final Node<T, S> child = context.selector().select(entry.geometry().mbr(), children);
63          List<Node<T, S>> list = child.add(entry);
64          List<? extends Node<T, S>> children2 = Util.replace(children, child, list);
65          if (children2.size() <= context.maxChildren())
66              return Collections.singletonList((Node<T, S>) new NonLeaf<T, S>(children2, context));
67          else {
68              ListPair<? extends Node<T, S>> pair = context.splitter().split(children2,
69                      context.minChildren());
70              return makeNonLeaves(pair);
71          }
72      }
73  
74      private List<Node<T, S>> makeNonLeaves(ListPair<? extends Node<T, S>> pair) {
75          List<Node<T, S>> list = new ArrayList<Node<T, S>>();
76          list.add(new NonLeaf<T, S>(pair.group1().list(), context));
77          list.add(new NonLeaf<T, S>(pair.group2().list(), context));
78          return list;
79      }
80  
81      @Override
82      public NodeAndEntries<T, S> delete(Entry<? extends T, ? extends S> entry, boolean all) {
83          // the result of performing a delete of the given entry from this node
84          // will be that zero or more entries will be needed to be added back to
85          // the root of the tree (because num entries of their node fell below
86          // minChildren),
87          // zero or more children will need to be removed from this node,
88          // zero or more nodes to be added as children to this node(because
89          // entries have been deleted from them and they still have enough
90          // members to be active)
91          List<Entry<T, S>> addTheseEntries = new ArrayList<Entry<T, S>>();
92          List<Node<T, S>> removeTheseNodes = new ArrayList<Node<T, S>>();
93          List<Node<T, S>> addTheseNodes = new ArrayList<Node<T, S>>();
94          int countDeleted = 0;
95  
96          for (final Node<T, S> child : children) {
97              if (entry.geometry().intersects(child.geometry().mbr())) {
98                  final NodeAndEntries<T, S> result = child.delete(entry, all);
99                  if (result.node().isPresent()) {
100                     if (result.node().get() != child) {
101                         // deletion occurred and child is above minChildren so
102                         // we update it
103                         addTheseNodes.add(result.node().get());
104                         removeTheseNodes.add(child);
105                         addTheseEntries.addAll(result.entriesToAdd());
106                         countDeleted += result.countDeleted();
107                         if (!all)
108                             break;
109                     }
110                     // else nothing was deleted from that child
111                 } else {
112                     // deletion occurred and brought child below minChildren
113                     // so we redistribute its entries
114                     removeTheseNodes.add(child);
115                     addTheseEntries.addAll(result.entriesToAdd());
116                     countDeleted += result.countDeleted();
117                     if (!all)
118                         break;
119                 }
120             }
121         }
122         if (removeTheseNodes.isEmpty())
123             return new NodeAndEntries<T, S>(of(this), Collections.<Entry<T, S>> emptyList(), 0);
124         else {
125             List<Node<T, S>> nodes = Util.remove(children, removeTheseNodes);
126             nodes.addAll(addTheseNodes);
127             if (nodes.size() == 0)
128                 return new NodeAndEntries<T, S>(Optional.<Node<T, S>> absent(), addTheseEntries,
129                         countDeleted);
130             else {
131                 NonLeaf<T, S> node = new NonLeaf<T, S>(nodes, context);
132                 return new NodeAndEntries<T, S>(of(node), addTheseEntries, countDeleted);
133             }
134         }
135     }
136 }
