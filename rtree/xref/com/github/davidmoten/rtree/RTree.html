1   package com.github.davidmoten.rtree;
2   
3   import rx.Observable;
4   
5   import com.github.davidmoten.util.ImmutableStack;
6   import com.google.common.base.Optional;
7   import com.google.common.collect.Lists;
8   
9   public class RTree {
10  
11  	private final Optional<Node> root;
12  	private final Context context;
13  
14  	private RTree(Optional<Node> root, Context context) {
15  		this.root = root;
16  		this.context = context;
17  	}
18  
19  	private RTree(Node root, Context context) {
20  		this(Optional.of(root), context);
21  	}
22  
23  	public RTree() {
24  		this(Optional.<Node> absent(), Context.DEFAULT);
25  	}
26  
27  	public RTree add(Entry entry) {
28  		if (root.isPresent())
29  			return new RTree(root.get().add(entry,
30  					ImmutableStack.<NonLeaf> empty()), context);
31  		else
32  			return new RTree(new Leaf(Lists.newArrayList(entry), context),
33  					context);
34  	}
35  
36  	public Observable<Entry> search(Rectangle r) {
37  		if (root.isPresent())
38  			return Observable.create(new OnSubscribeSearch(root.get(), r));
39  		else
40  			return Observable.empty();
41  	}
42  
43  	public Observable<Entry> entries() {
44  		if (root.isPresent())
45  			return Observable.create(new OnSubscribeEntries(root.get()));
46  		else
47  			return Observable.empty();
48  	}
49  
50  	public Visualizer visualize(int width, int height, Rectangle view,
51  			int maxDepth) {
52  		return new Visualizer(this, width, height, view);
53  	}
54  
55  	public Optional<Node> root() {
56  		return root;
57  	}
58  
59  }
