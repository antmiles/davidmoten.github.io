10  
11  public class SelectorMinimalAreaIncrease implements Selector {
12  
13      @Override
14      public <T> Node<T> select(Geometry g, List<? extends Node<T>> nodes) {
15          return findLeastIncreaseInMbrArea(g.mbr(), nodes);
16      }
17  
18      private static <T> Node<T> findLeastIncreaseInMbrArea(Rectangle r, List<? extends Node<T>> list) {
19          Optional<Double> minDifference = Optional.absent();
20          Optional<Node<T>> minDiffItem = Optional.absent();
21          for (Node<T> m : list) {
22              double diff = m.geometry().mbr().add(r).area() - m.geometry().mbr().area();
23              if (!minDifference.isPresent() || diff < minDifference.get()) {
24                  minDifference = of(diff);
25                  minDiffItem = of(m);
26              }
27          }
28          return minDiffItem.get();
29      }

31  }
