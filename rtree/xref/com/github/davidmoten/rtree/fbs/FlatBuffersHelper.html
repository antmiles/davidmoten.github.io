36      static <T, S extends Geometry> int addEntries(List<Entry<T, S>> entries,
37              FlatBufferBuilder builder, Func1<? super T, byte[]> serializer) {
38          int[] entries2 = new int[entries.size()];
39          for (int i = 0; i < entries.size(); i++) {
40              Geometry g = entries.get(i).geometry();
41              final int geom;
42              final byte geomType;
43              if (g instanceof Rectangle) {
44                  Rectangle b = (Rectangle) g;
45                  geom = Box_.createBox_(builder, b.x1(), b.y1(), b.x2(), b.y2());
46                  geomType = GeometryType_.Box;
47              } else if (g instanceof Point) {
48                  Point p = (Point) g;
49                  geom = Point_.createPoint_(builder, p.x(), p.y());
50                  geomType = GeometryType_.Point;
51              } else if (g instanceof Circle) {
52                  Circle c = (Circle) g;
53                  geom = Circle_.createCircle_(builder, c.x(), c.y(), c.radius());
54                  geomType = GeometryType_.Circle;
55              } else if (g instanceof Line) {
56                  Line c = (Line) g;
57                  geom = Line_.createLine_(builder, c.x1(), c.y1(), c.x2(), c.y2());
58                  geomType = GeometryType_.Line;
59              } else
60                  throw new RuntimeException("unexpected");
61  
62              Geometry_.startGeometry_(builder);
63              if (geomType == GeometryType_.Box) {
64                  Geometry_.addBox(builder, geom);
65              } else if (geomType == GeometryType_.Point) {
66                  Geometry_.addPoint(builder, geom);
67              } else if (geomType == GeometryType_.Circle) {
68                  Geometry_.addCircle(builder, geom);
69              } else if (geomType == GeometryType_.Line) {
70                  Geometry_.addLine(builder, geom);
71              } else
72                  throw new RuntimeException("unexpected");
73  
74              Geometry_.addType(builder, geomType);
75              int geo = Geometry_.endGeometry_(builder);
76              int obj = Entry_.createObjectVector(builder, serializer.call(entries.get(i).value()));
77              entries2[i] = Entry_.createEntry_(builder, geo, obj);
78          }
79  
80          int ents = Node_.createEntriesVector(builder, entries2);
81          Rectangle mbb = Util.mbr(entries);
82          int b = Box_.createBox_(builder, mbb.x1(), mbb.y1(), mbb.x2(), mbb.y2());
83          Node_.startNode_(builder);
84          Node_.addMbb(builder, b);
85          Node_.addEntries(builder, ents);
86          return Node_.endNode_(builder);
87  
88      }

90      static <T, S extends Geometry> List<Entry<T, S>> createEntries(Node_ node,
91              Func1<byte[], ? extends T> deserializer) {
92          List<Entry<T, S>> entries = new ArrayList<Entry<T, S>>();
93          int numEntries = node.entriesLength();
94          Preconditions.checkArgument(numEntries > 0);
95          Entry_ entry = new Entry_();
96          Geometry_ geom = new Geometry_();
97          for (int i = 0; i < numEntries; i++) {
98              Entry<T, S> ent = createEntry(node, deserializer, entry, geom, i);
99              entries.add(ent);
100         }
101         return entries;
102     }

104     @SuppressWarnings("unchecked")
105     private static <T, S extends Geometry> Entry<T, S> createEntry(Node_ node,
106             Func1<byte[], ? extends T> deserializer, Entry_ entry, Geometry_ geom, int i) {
107         node.entries(entry, i);
108         entry.geometry(geom);
109         final Geometry g = toGeometry(geom);
110         return Entries.entry(parseObject(deserializer, entry), (S) g);
111     }

113     static <T, S extends Geometry> Entry<T, S> createEntry(Node_ node,
114             Func1<byte[], ? extends T> deserializer, int i) {
115         return createEntry(node, deserializer, new Entry_(), new Geometry_(), i);
116     }

118     static <T> T parseObject(Func1<byte[], ? extends T> deserializer, Entry_ entry) {
119         ByteBuffer bb = entry.objectAsByteBuffer();
120         byte[] bytes = Arrays.copyOfRange(bb.array(), bb.position(), bb.limit());
121         T t = deserializer.call(bytes);
122         return t;
123     }

125     @SuppressWarnings("unchecked")
126     static <S extends Geometry> S toGeometry(Geometry_ g) {
127         final Geometry result;
128         byte type = g.type();
129         if (type == GeometryType_.Box) {
130             result = createBox(g.box());
131         } else if (type == GeometryType_.Point) {
132             result = Geometries.point(g.point().x(), g.point().y());
133         } else if (type == GeometryType_.Circle) {
134             Circle_ c = g.circle();
135             result = Geometries.circle(c.x(), c.y(), c.radius());
136         } else if (type == GeometryType_.Line) {
137             result = createLine(g.line());
138         } else
139             throw new RuntimeException("unexpected");
140         return (S) result;
141     }

143     static Rectangle createBox(Box_ b) {
144         return Geometries.rectangle(b.minX(), b.minY(), b.maxX(), b.maxY());
145     }

147     static Line createLine(Box_ b) {
148         return Geometries.line(b.minX(), b.minY(), b.maxX(), b.maxY());
