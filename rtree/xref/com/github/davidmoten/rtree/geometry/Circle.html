3   public class Circle implements Geometry {
4   
5       private final float x, y, radius;
6       private final Rectangle mbr;
7   
8       public Circle(float x, float y, float radius) {
9           this.x = x;
10          this.y = y;
11          this.radius = radius;
12          this.mbr = Rectangle.create(x - radius, y - radius, x + radius, y + radius);
13      }
14  
15      public static Circle create(double x, double y, double radius) {
16          return new Circle((float) x, (float) y, (float) radius);
17      }
18  
19      @Override
20      public Rectangle mbr() {
21          return mbr;
22      }
23  
24      @Override
25      public double distance(Rectangle r) {
26          return Math.max(0, new Point(x, y).distance(r) - radius);
27      }
28  
29      @Override
30      public boolean intersects(Rectangle r) {
31          return new Point(x, y).distance(r) <= radius;
32      }
33  
34      @Override
35      public int hashCode() {
36          final int prime = 31;
37          int result = 1;
38          result = prime * result + Float.floatToIntBits(radius);
39          result = prime * result + Float.floatToIntBits(x);
40          result = prime * result + Float.floatToIntBits(y);
41          return result;
42      }
43  
44      @Override
45      public boolean equals(Object obj) {
46          if (this == obj)
47              return true;
48          if (obj == null)
49              return false;
50          if (getClass() != obj.getClass())
51              return false;
52          Circle other = (Circle) obj;
53          if (Float.floatToIntBits(radius) != Float.floatToIntBits(other.radius))
54              return false;
55          if (Float.floatToIntBits(x) != Float.floatToIntBits(other.x))
56              return false;
57          if (Float.floatToIntBits(y) != Float.floatToIntBits(other.y))
58              return false;
59          return true;
60      }
