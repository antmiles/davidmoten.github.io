23      public static Rectangle rectangleGeographic(double lon1, double lat1, double lon2,
24              double lat2) {
25          double x1 = normalizeLongitude(lon1);
26          double x2 = normalizeLongitude(lon2);
27          if (x2 < x1) {
28              x2 += 360;
29          }
30          return rectangle(x1, lat1, x2, lat2);
31      }
32  
33      public static Point pointGeographic(double lon, double lat) {
34          return point(normalizeLongitude(lon), lat);
35      }
36  
37      @VisibleForTesting
38      static double normalizeLongitude(double d) {
39          if (d == -180.0)
40              return -180.0;
41          else {
42              double sign = Math.signum(d);
43              double x = Math.abs(d) / 360;
44              double x2 = (x - Math.floor(x)) * 360;
45              if (x2 >= 180)
46                  x2 -= 360;
47              return x2 * sign;
48          }
49      }
50  
51      public static Line line(double x1, double y1, double x2, double y2) {
52          return Line.create((float) x1, (float) y1, (float) x2, (float) y2);
53      }
