3   import java.lang.reflect.Method;
4   import java.lang.reflect.Modifier;
5   import java.util.Map.Entry;
6   import java.util.TreeMap;
7   
8   import rx.Observable;
9   
10  public class MethodListerMain {
11  
12      public static void main(String[] args) {
13          TreeMap<String, String> map = new TreeMap<String, String>();
14          for (Method method : Observable.class.getMethods()) {
15              if (Modifier.isPublic(method.getModifiers())
16                      && !Modifier.isNative(method.getModifiers()))
17                  map.put(method.getName(), method.toString());
18          }
19          System.out.println("| Method | Signature |");
20          System.out.println("| ------ | --------- |");
21          for (Entry<String, String> entry : map.entrySet()) {
22              System.out
23                      .println("| " + entry.getKey() + " | "
24                              + entry.getValue().replaceAll("\\w+\\.", "").replaceAll("public ", "")
25                                      .replaceAll("final ", "").replaceAll("Observable", "Ob")
26                              + " |");
27          }
28      }
29  }
