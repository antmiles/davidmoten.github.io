22  /**
23   * Provides builder for the {@link Func1} parameter of
24   * {@link Observable#retryWhen(Func1)}. For example:
25   * 
26   * <pre>
27   * o.retryWhen(RetryWhen.maxRetries(4).delay(10, TimeUnit.SECONDS).action(log).build());
28   * </pre>
29   * 
30   * <p>
31   * or
32   * </p>
33   * 
34   * <pre>
35   * o.retryWhen(RetryWhen.exponentialBackoff(100, TimeUnit.MILLISECONDS).maxRetries(10).build());
36   * </pre>
37   */
38  public final class RetryWhen {
39  
40      private static final long NO_MORE_DELAYS = -1;
41  
42      private static Func1<Observable<? extends Throwable>, Observable<?>> notificationHandler(
43              final Observable<Long> delays, final Scheduler scheduler,
44              final Action1<? super ErrorAndDuration> action,
45              final List<Class<? extends Throwable>> retryExceptions,
46              final List<Class<? extends Throwable>> failExceptions,
47              final Func1<? super Throwable, Boolean> exceptionPredicate) {
48  
49          final Func1<ErrorAndDuration, Observable<ErrorAndDuration>> checkExceptions = createExceptionChecker(
50                  retryExceptions, failExceptions, exceptionPredicate);
51  
52          return createNotificationHandler(delays, scheduler, action, checkExceptions);
53      }
54  
55      private static Func1<Observable<? extends Throwable>, Observable<?>> createNotificationHandler(
56              final Observable<Long> delays, final Scheduler scheduler,
57              final Action1<? super ErrorAndDuration> action,
58              final Func1<ErrorAndDuration, Observable<ErrorAndDuration>> checkExceptions) {
59          return new Func1<Observable<? extends Throwable>, Observable<?>>() {
60  
61              @Override
62              public Observable<ErrorAndDuration> call(Observable<? extends Throwable> errors) {
63                  return errors
64                          // zip with delays, use -1 to signal completion
65                          .zipWith(delays.concatWith(just(NO_MORE_DELAYS)), TO_ERROR_AND_DURATION)
66                          // check retry and non-retry exceptions
67                          .flatMap(checkExceptions)
68                          // perform user action (for example log that a
69                          // delay is happening)
70                          .doOnNext(callActionExceptForLast(action))
71                          // delay the time in ErrorAndDuration
72                          .flatMap(delay(scheduler));
73              }
74          };
75      }
76  
77      private static Action1<ErrorAndDuration> callActionExceptForLast(
78              final Action1<? super ErrorAndDuration> action) {
79          return new Action1<ErrorAndDuration>() {
80  
81              @Override
82              public void call(ErrorAndDuration e) {
83                  if (e.durationMs() != NO_MORE_DELAYS)
84                      action.call(e);
85              }
86  
87          };
88      }
89  
90      // TODO unit test
91      private static Func1<ErrorAndDuration, Observable<ErrorAndDuration>> createExceptionChecker(
92              final List<Class<? extends Throwable>> retryExceptions,
93              final List<Class<? extends Throwable>> failExceptions,
94              final Func1<? super Throwable, Boolean> exceptionPredicate) {
95          return new Func1<ErrorAndDuration, Observable<ErrorAndDuration>>() {
96  
97              @Override
98              public Observable<ErrorAndDuration> call(ErrorAndDuration e) {
99                  if (!exceptionPredicate.call(e.throwable()))
100                     return Observable.error(e.throwable());
101                 for (Class<? extends Throwable> cls : failExceptions) {
102                     if (e.throwable().getClass().isAssignableFrom(cls))
103                         return Observable.error(e.throwable());
104                 }
105                 if (retryExceptions.size() > 0) {
106                     for (Class<? extends Throwable> cls : retryExceptions) {
107                         if (e.throwable().getClass().isAssignableFrom(cls))
108                             return just(e);
109                     }
110                     return Observable.error(e.throwable());
111                 } else {
112                     return just(e);
113                 }
114             }
115         };
116     }
117 
118     private static Func2<Throwable, Long, ErrorAndDuration> TO_ERROR_AND_DURATION = new Func2<Throwable, Long, ErrorAndDuration>() {
119         @Override
120         public ErrorAndDuration call(Throwable throwable, Long durationMs) {
121             return new ErrorAndDuration(throwable, durationMs);
122         }
123     };
124 
125     private static Func1<ErrorAndDuration, Observable<ErrorAndDuration>> delay(
126             final Scheduler scheduler) {
127         return new Func1<ErrorAndDuration, Observable<ErrorAndDuration>>() {
128             @Override
129             public Observable<ErrorAndDuration> call(ErrorAndDuration e) {
130                 if (e.durationMs() == NO_MORE_DELAYS)
131                     return Observable.error(e.throwable());
132                 else
133                     return Observable.timer(e.durationMs(), TimeUnit.MILLISECONDS, scheduler)
134                             .map(Functions.constant(e));
135             }
136         };
137     }

182     public static final class Builder {
183 
184         private final List<Class<? extends Throwable>> retryExceptions = new ArrayList<Class<? extends Throwable>>();
185         private final List<Class<? extends Throwable>> failExceptions = new ArrayList<Class<? extends Throwable>>();
186         private Func1<? super Throwable, Boolean> exceptionPredicate = Functions.alwaysTrue();
187 
188         private Observable<Long> delays = Observable.just(0L).repeat();
189         private Optional<Integer> maxRetries = absent();
190         private Optional<Scheduler> scheduler = of(Schedulers.computation());
191         private Action1<? super ErrorAndDuration> action = Actions.doNothing1();
192 
193         private Builder() {
194             // must use static factory method to instantiate
195         }
196 
197         public Builder retryWhenInstanceOf(Class<? extends Throwable>... classes) {
198             retryExceptions.addAll(Arrays.asList(classes));
199             return this;
200         }
201 
202         public Builder failWhenInstanceOf(Class<? extends Throwable>... classes) {
203             failExceptions.addAll(Arrays.asList(classes));
204             return this;
205         }
206 
207         public Builder retryIf(Func1<Throwable, Boolean> predicate) {
208             this.exceptionPredicate = predicate;
209             return this;
210         }
211 
212         public Builder delays(Observable<Long> delays, TimeUnit unit) {
213             this.delays = delays.map(toMillis(unit));
214             return this;
215         }
216 
217         public Builder delay(Long delay, final TimeUnit unit) {
218             this.delays = Observable.just(delay).map(toMillis(unit)).repeat();
219             return this;
220         }
221 
222         private static Func1<Long, Long> toMillis(final TimeUnit unit) {
223             return new Func1<Long, Long>() {
224 
225                 @Override
226                 public Long call(Long t) {
227                     return unit.toMillis(t);
228                 }
229             };
230         }
231 
232         public Builder maxRetries(int maxRetries) {
233             this.maxRetries = of(maxRetries);
234             return this;
235         }
236 
237         public Builder scheduler(Scheduler scheduler) {
238             this.scheduler = of(scheduler);
239             return this;
240         }
241 
242         public Builder action(Action1<? super ErrorAndDuration> action) {
243             this.action = action;
244             return this;
245         }
246 
247         public Builder exponentialBackoff(final long firstDelay, final TimeUnit unit,
248                 final double factor) {
249             delays = Observable.range(1, Integer.MAX_VALUE)
250                     // make exponential
251                     .map(new Func1<Integer, Long>() {
252                         @Override
253                         public Long call(Integer n) {
254                             return Math.round(Math.pow(factor, n - 1) * unit.toMillis(firstDelay));
255                         }
256                     });
257             return this;
258         }
259 
260         public Builder exponentialBackoff(long firstDelay, TimeUnit unit) {
261             return exponentialBackoff(firstDelay, unit, 2);
262         }
263 
264         public Func1<Observable<? extends Throwable>, Observable<?>> build() {
265             Preconditions.checkNotNull(delays);
266             if (maxRetries.isPresent()) {
267                 delays = delays.take(maxRetries.get());
268             }
269             return notificationHandler(delays, scheduler.get(), action, retryExceptions,
270                     failExceptions, exceptionPredicate);
271         }
