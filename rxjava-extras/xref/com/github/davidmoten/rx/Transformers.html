154      * Returns a {@link Transformer} that allows processing of the source stream
155      * to be defined in a state machine where transitions of the state machine
156      * may also emit items to downstream that are buffered if necessary when
157      * backpressure is requested. <code>flatMap</code> is part of the processing
158      * chain so the source may experience requests for more items than are
159      * strictly required by the endpoint subscriber.
160      * 
161      * @param initialStateFactory
162      *            the factory to create the initial state of the state machine.
163      * @param transition
164      *            defines state transitions and consequent emissions to
165      *            downstream when an item arrives from upstream. The
166      *            {@link Subscriber} is called with the emissions to downstream.
167      *            You can optionally call {@link Subscriber#isUnsubscribed()} to
168      *            check if you can stop emitting from the transition. If you do
169      *            wish to terminate the Observable then call
170      *            {@link Subscriber#unsubscribe()} and return anything (say
171      *            {@code null} from the transition (as the next state which will
172      *            not be used). You can also complete the Observable by calling
173      *            {@link Subscriber#onCompleted} or {@link Subscriber#onError}
174      *            from within the transition and return anything from the
175      *            transition (will not be used). The transition should run
176      *            synchronously so that completion of a call to the transition
177      *            should also signify all emissions from that transition have
178      *            been made.
179      * @param completion
180      *            defines activity that should happen based on the final state
181      *            just before downstream <code>onCompleted()</code> is called.
182      *            For example any buffered emissions in state could be emitted
183      *            at this point. Don't call <code>observer.onCompleted()</code>
184      *            as it is called for you after the action completes if and only
185      *            if you return true from this function.
186      * @param backpressureStrategy
187      *            is applied to the emissions from one call of transition and
188      *            should enforce backpressure.
189      * @param <State>
190      *            the class representing the state of the state machine
191      * @param <In>
192      *            the input observable type
193      * @param <Out>
194      *            the output observable type
195      * @throws NullPointerException
196      *             if {@code initialStateFactory} or {@code transition},or
197      *             {@code completionAction} is null
198      * @return a backpressure supporting transformer that implements the state
199      *         machine specified by the parameters
200      */

202             Func0<State> initialStateFactory,
203             Func3<? super State, ? super In, ? super Subscriber<Out>, ? extends State> transition,
204             Func2<? super State, ? super Subscriber<Out>, Boolean> completion,
205             BackpressureStrategy backpressureStrategy) {
206         return TransformerStateMachine.<State, In, Out> create(initialStateFactory, transition,
207                 completion, backpressureStrategy, DEFAULT_INITIAL_BATCH);
208     }

394      * Returns a {@link Transformer} that returns an {@link Observable} that is
395      * collected into {@code Collection} instances created by {@code factory}
396      * that are emitted when the collection and latest emission do not satisfy
397      * {@code condition} or on completion.
398      * 
399      * @param factory
400      *            collection instance creator
401      * @param collect
402      *            collection action
403      * @param condition
404      *            returns true if and only if emission should be collected in
405      *            current collection being prepared for emission
406      * @param isEmpty
407      *            indicates that the collection is empty
408      * @param <T>
409      *            generic type of source observable
410      * @param <R>
411      *            collection type emitted by transformed Observable
412      * @return transformer as above
413      */
414     public static <T, R> Transformer<T, R> collectWhile(final Func0<R> factory,
415             final Action2<? super R, ? super T> collect,
416             final Func2<? super R, ? super T, Boolean> condition,
417             final Func1<? super R, Boolean> isEmpty) {
418         Func3<R, T, Observer<R>, R> transition = new Func3<R, T, Observer<R>, R>() {
419 
420             @Override
421             public R call(R collection, T t, Observer<R> observer) {
422 
423                 if (condition.call(collection, t)) {
424                     collect.call(collection, t);
425                     return collection;
426                 } else {
427                     observer.onNext(collection);
428                     R r = factory.call();
429                     collect.call(r, t);
430                     return r;
431                 }
432             }
433 
434         };
435         Func2<R, Observer<R>, Boolean> completionAction = new Func2<R, Observer<R>, Boolean>() {
436             @Override
437             public Boolean call(R collection, Observer<R> observer) {
438                 if (!isEmpty.call(collection)) {
439                     observer.onNext(collection);
440                 }
441                 return true;
442             }
443         };
444         return Transformers.stateMachine(factory, transition, completionAction);
445     }

659     /**
660      * Groups the items emitted by an {@code Observable} according to a
661      * specified criterion, and emits these grouped items as
662      * {@link GroupedObservable}s. The emitted {@code GroupedObservable} allows
663      * only a single {@link Subscriber} during its lifetime and if this
664      * {@code Subscriber} unsubscribes before the source terminates, the next
665      * emission by the source having the same key will trigger a new
666      * {@code GroupedObservable} emission.
667      * <p>
668      * <img width="640" height="360" src=
669      * "https://raw.github.com/wiki/ReactiveX/RxJava/images/rx-operators/groupBy.png"
670      * alt="">
671      * <p>
672      * <em>Note:</em> A {@link GroupedObservable} will cache the items it is to
673      * emit until such time as it is subscribed to. For this reason, in order to
674      * avoid memory leaks, you should not simply ignore those
675      * {@code GroupedObservable}s that do not concern you. Instead, you can
676      * signal to them that they may discard their buffers by applying an
677      * operator like {@code .ignoreElements()} to them.
678      * <dl>
679      * <dt><b>Scheduler:</b></dt>
680      * <dd>{@code groupBy} does not operate by default on a particular
681      * {@link Scheduler}.</dd>
682      * </dl>
683      * 
684      * @param keySelector
685      *            a function that extracts the key for each item
686      * @param elementSelector
687      *            a function that extracts the return element for each item
688      * @param evictingMapFactory
689      *            a function that given an eviction action returns a {@link Map}
690      *            instance that will be used to assign items to the appropriate
691      *            {@code GroupedObservable}s. The {@code Map} instance must be
692      *            thread-safe and any eviction must trigger a call to the
693      *            supplied action (synchronously or asynchronously). This can be
694      *            used to limit the size of the map by evicting keys by maximum
695      *            size or access time for instance. If
696      *            {@code evictingMapFactory} is null then no eviction strategy
697      *            will be applied (and a suitable default thread-safe
698      *            implementation of {@code Map} will be supplied). Here's an
699      *            example using Guava's {@code CacheBuilder} from v19.0:
700      * 
701      *            <pre>
702      *            {@code
703      *            Func1<Action1<K>, Map<K, Object>> mapFactory 
704      *              = action -> CacheBuilder.newBuilder()
705      *                  .maximumSize(1000)
706      *                  .expireAfterAccess(12, TimeUnit.HOUR)
707      *                  .removalListener(key -> action.call(key))
708      *                  .<K, Object> build().asMap();
709      *            }
710      *            </pre>
711      * 
712      * @param <T>
713      *            the type of the input observable
714      * @param <K>
715      *            the key type
716      * @param <R>
717      *            the element type
718      * @return an {@code Observable} that emits {@link GroupedObservable}s, each
719      *         of which corresponds to a unique key value and each of which
720      *         emits those items from the source Observable that share that key
721      *         value
722      * @see <a href="http://reactivex.io/documentation/operators/groupby.html">
723      *      ReactiveX operators documentation: GroupBy</a>
724      */
725     public static <T, K, R> Transformer<T, GroupedObservable<K, R>> groupByEvicting(
726             final Func1<? super T, ? extends K> keySelector,
727             final Func1<? super T, ? extends R> elementSelector,
728             final Func1<Action1<K>, Map<K, Object>> evictingMapFactory) {
729         return new Transformer<T, GroupedObservable<K, R>>() {
730 
731             @Override
732             public Observable<GroupedObservable<K, R>> call(Observable<T> o) {
733                 return o.lift(
734                         new com.github.davidmoten.rx.internal.operators.OperatorGroupBy<T, K, R>(
735                                 keySelector, elementSelector, evictingMapFactory));
736             }
737         };
738     }

785     /**
786      * Removes pairs non-recursively from a stream. Uses
787      * {@code Transformers.stateMachine()} under the covers to ensure items are
788      * emitted as soon as possible (if an item can't be in a pair then it is
789      * emitted straight away).
790      * 
791      * @param isCandidateForFirst
792      *            returns true if item is potentially the first of a pair that
793      *            we want to remove
794      * @param remove
795      *            returns true if a pair should be removed
796      * @param <T>
797      *            generic type of stream being transformed
798      * @return transformed stream
799      */
800     public static <T> Transformer<T, T> removePairs(
801             final Func1<? super T, Boolean> isCandidateForFirst,
802             final Func2<? super T, ? super T, Boolean> remove) {
803         return new Transformer<T, T>() {
804 
805             @Override
806             public Observable<T> call(Observable<T> o) {
807                 return o.compose(Transformers. //
808                         stateMachine() //
809                         .initialState(Optional.<T> absent()) //
810                         .transition(new Transition<Optional<T>, T, T>() {
811 
812                     @Override
813                     public Optional<T> call(Optional<T> state, T value, Subscriber<T> subscriber) {
814                         if (!state.isPresent()) {
815                             if (isCandidateForFirst.call(value)) {
816                                 return Optional.of(value);
817                             } else {
818                                 subscriber.onNext(value);
819                                 return Optional.absent();
820                             }
821                         } else {
822                             if (remove.call(state.get(), value)) {
823                                 // emit nothing and reset state
824                                 return Optional.absent();
825                             } else {
826                                 subscriber.onNext(state.get());
827                                 if (isCandidateForFirst.call(value)) {
828                                     return Optional.of(value);
829                                 } else {
830                                     subscriber.onNext(value);
831                                     return Optional.absent();
832                                 }
833                             }
834                         }
835                     }
836                 }).completion(new Completion<Optional<T>, T>() {
837 
838                     @Override
839                     public Boolean call(Optional<T> state, Subscriber<T> subscriber) {
840                         if (state.isPresent())
841                             subscriber.onNext(state.get());
842                         // yes, complete
843                         return true;
844                     }
845                 }).build());
846             }
847         };
848     }
