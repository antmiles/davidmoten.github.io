33  /**
34   * Groups the items emitted by an Observable according to a specified criterion,
35   * and emits these grouped items as Observables, one Observable per group.
36   * <p>
37   * <img width="640" height="360" src=
38   * "https://raw.githubusercontent.com/wiki/ReactiveX/RxJava/images/rx-operators/groupBy.png"
39   * alt="">
40   *
41   * @param <K>
42   *            the key type
43   * @param <T>
44   *            the source and group value type
45   * @param <V>
46   *            the value type of the groups
47   */
48  public final class OperatorGroupBy<T, K, V> implements Operator<GroupedObservable<K, V>, T> {
49      final Func1<? super T, ? extends K> keySelector;
50      final Func1<? super T, ? extends V> valueSelector;
51      final int bufferSize;
52      final boolean delayError;
53      final Func1<Action1<K>, Map<K, Object>> mapFactory; // nullable
54  
55      @SuppressWarnings({ "unchecked", "rawtypes" })
56      public OperatorGroupBy(Func1<? super T, ? extends K> keySelector) {
57          this(keySelector, (Func1) UtilityFunctions.<T> identity(), RxRingBuffer.SIZE, false, null);
58      }

61              Func1<? super T, ? extends V> valueSelector) {

66              Func1<? super T, ? extends V> valueSelector,

71      public OperatorGroupBy(Func1<? super T, ? extends K> keySelector,
72              Func1<? super T, ? extends V> valueSelector, int bufferSize, boolean delayError,
73              Func1<Action1<K>, Map<K, Object>> mapFactory) {
74          this.keySelector = keySelector;
75          this.valueSelector = valueSelector;
76          this.bufferSize = bufferSize;
77          this.delayError = delayError;
78          this.mapFactory = mapFactory;
79      }

81      @Override
82      public Subscriber<? super T> call(Subscriber<? super GroupedObservable<K, V>> child) {
83          final GroupBySubscriber<T, K, V> parent;
84          try {
85              parent = new GroupBySubscriber<T, K, V>(child, keySelector, valueSelector, bufferSize,
86                      delayError, mapFactory);
87          } catch (Throwable ex) {
88              // Can reach here because mapFactory.call() may throw in constructor
89              // of GroupBySubscriber
90              Exceptions.throwOrReport(ex, child);
91              Subscriber<? super T> parent2 = Subscribers.empty();
92              parent2.unsubscribe();
93              return parent2;
94          }
95  
96          child.add(Subscriptions.create(new Action0() {
97              @Override
98              public void call() {
99                  parent.cancel();
100             }
101         }));
102 
103         child.setProducer(parent.producer);
104 
105         return parent;
106     }

121     public static final class GroupBySubscriber<T, K, V> extends Subscriber<T> {
122         final Subscriber<? super GroupedObservable<K, V>> actual;
123         final Func1<? super T, ? extends K> keySelector;
124         final Func1<? super T, ? extends V> valueSelector;
125         final int bufferSize;
126         final boolean delayError;
127         final Map<Object, GroupedUnicast<K, V>> groups;
128         final Queue<GroupedObservable<K, V>> queue;
129         final GroupByProducer producer;
130         final Queue<K> evictedKeys;
131 
132         static final Object NULL_KEY = new Object();
133 
134         final ProducerArbiter s;
135 
136         volatile int cancelled;
137         @SuppressWarnings("rawtypes")
138         static final AtomicIntegerFieldUpdater<GroupBySubscriber> CANCELLED = AtomicIntegerFieldUpdater
139                 .newUpdater(GroupBySubscriber.class, "cancelled");
140 
141         volatile long requested;
142         @SuppressWarnings("rawtypes")
143         static final AtomicLongFieldUpdater<GroupBySubscriber> REQUESTED = AtomicLongFieldUpdater
144                 .newUpdater(GroupBySubscriber.class, "requested");
145 
146         volatile int groupCount;
147         @SuppressWarnings("rawtypes")
148         static final AtomicIntegerFieldUpdater<GroupBySubscriber> GROUP_COUNT = AtomicIntegerFieldUpdater
149                 .newUpdater(GroupBySubscriber.class, "groupCount");
150 
151         Throwable error;
152         volatile boolean done;
153 
154         volatile int wip;
155         @SuppressWarnings("rawtypes")
156         static final AtomicIntegerFieldUpdater<GroupBySubscriber> WIP = AtomicIntegerFieldUpdater
157                 .newUpdater(GroupBySubscriber.class, "wip");

159         public GroupBySubscriber(Subscriber<? super GroupedObservable<K, V>> actual,
160                 Func1<? super T, ? extends K> keySelector,
161                 Func1<? super T, ? extends V> valueSelector, int bufferSize, boolean delayError,
162                 Func1<Action1<K>, Map<K, Object>> mapFactory) {
163             this.actual = actual;
164             this.keySelector = keySelector;
165             this.valueSelector = valueSelector;
166             this.bufferSize = bufferSize;
167             this.delayError = delayError;
168             this.queue = new ConcurrentLinkedQueue<GroupedObservable<K, V>>();
169             GROUP_COUNT.lazySet(this, 1);
170             this.s = new ProducerArbiter();
171             this.s.request(bufferSize);
172             this.producer = new GroupByProducer(this);
173             if (mapFactory == null) {
174                 this.groups = new ConcurrentHashMap<Object, GroupedUnicast<K, V>>();
175                 this.evictedKeys = null;
176             } else {
177                 this.evictedKeys = new ConcurrentLinkedQueue<K>();
178                 this.groups = createMap(mapFactory, new EvictionAction<K>(evictedKeys));
179             }
180         }
181 
182         // declare a class instead of using anonymous class to
183         // limit enclosing scope. Is this of value?
184         static class EvictionAction<K> implements Action1<K> {
185 
186             final Queue<K> evictedKeys;
187 
188             EvictionAction(Queue<K> evictedKeys) {
189                 this.evictedKeys = evictedKeys;
190             }
191 
192             @Override
193             public void call(K key) {
194                 evictedKeys.offer(key);
195             }
196         }
197 
198         @SuppressWarnings("unchecked")
199         private Map<Object, GroupedUnicast<K, V>> createMap(
200                 Func1<Action1<K>, Map<K, Object>> mapFactory, Action1<K> evictionAction) {
201             return (Map<Object, GroupedUnicast<K, V>>) (Map<Object, ?>) mapFactory
202                     .call(evictionAction);
203         }
204 
205         @Override
206         public void setProducer(Producer s) {
207             this.s.setProducer(s);
208         }
209 
210         @Override
211         public void onNext(T t) {
212             if (done) {
213                 return;
214             }
215 
216             final Queue<GroupedObservable<K, V>> q = this.queue;
217             final Subscriber<? super GroupedObservable<K, V>> a = this.actual;
218 
219             K key;
220             try {
221                 key = keySelector.call(t);
222             } catch (Throwable ex) {
223                 unsubscribe();
224                 errorAll(a, q, ex);
225                 return;
226             }
227 
228             boolean notNew = true;
229             Object mapKey = key != null ? key : NULL_KEY;
230             GroupedUnicast<K, V> group = groups.get(mapKey);
231             if (group == null) {
232                 // if the main has been cancelled, stop creating groups
233                 // and skip this value
234                 if (cancelled == 0) {
235                     group = GroupedUnicast.createWith(key, bufferSize, this, delayError);
236                     groups.put(mapKey, group);
237 
238                     GROUP_COUNT.getAndIncrement(this);
239 
240                     notNew = false;
241                     q.offer(group);
242                     drain();
243                 } else {
244                     return;
245                 }
246             }
247 
248             V v;
249             try {
250                 v = valueSelector.call(t);
251             } catch (Throwable ex) {
252                 unsubscribe();
253                 errorAll(a, q, ex);
254                 return;
255             }
256 
257             group.onNext(v);
258 
259             if (evictedKeys != null) {
260                 K evictedKey;
261                 while ((evictedKey = evictedKeys.poll()) != null) {
262                     GroupedUnicast<K, V> g = groups.get(evictedKey);
263                     if (g != null) {
264                         g.onComplete();
265                     }
266                 }
267             }
268 
269             if (notNew) {
270                 s.request(1);
271             }
272         }
273 
274         @Override
275         public void onError(Throwable t) {
276             if (done) {
277                 RxJavaPlugins.getInstance().getErrorHandler().handleError(t);
278                 return;
279             }
280             error = t;
281             done = true;
282             GROUP_COUNT.decrementAndGet(this);
283             drain();
284         }
285 
286         @Override
287         public void onCompleted() {
288             if (done) {
289                 return;
290             }
291 
292             for (GroupedUnicast<K, V> e : groups.values()) {
293                 e.onComplete();
294             }
295             groups.clear();
296             if (evictedKeys != null) {
297                 evictedKeys.clear();
298             }
299 
300             done = true;
301             GROUP_COUNT.decrementAndGet(this);
302             drain();
303         }
304 
305         public void requestMore(long n) {
306             if (n < 0) {
307                 throw new IllegalArgumentException("n >= 0 required but it was " + n);
308             }
309 
310             BackpressureUtils.getAndAddRequest(REQUESTED, this, n);
311             drain();
312         }
313 
314         public void cancel() {
315             // cancelling the main source means we don't want any more groups
316             // but running groups still require new values
317             if (CANCELLED.compareAndSet(this, 0, 1)) {
318                 if (GROUP_COUNT.decrementAndGet(this) == 0) {
319                     unsubscribe();
320                 }
321             }
322         }
323 
324         public void cancel(K key) {
325             Object mapKey = key != null ? key : NULL_KEY;
326             if (groups.remove(mapKey) != null) {
327                 if (GROUP_COUNT.decrementAndGet(this) == 0) {
328                     unsubscribe();
329                 }
330             }
331         }
332 
333         void drain() {
334             if (WIP.getAndIncrement(this) != 0) {
335                 return;
336             }
337 
338             int missed = 1;
339 
340             final Queue<GroupedObservable<K, V>> q = this.queue;
341             final Subscriber<? super GroupedObservable<K, V>> a = this.actual;
342 
343             for (;;) {
344 
345                 if (checkTerminated(done, q.isEmpty(), a, q)) {
346                     return;
347                 }
348 
349                 long r = requested;
350                 boolean unbounded = r == Long.MAX_VALUE;
351                 long e = 0L;
352 
353                 while (r != 0) {
354                     boolean d = done;
355 
356                     GroupedObservable<K, V> t = q.poll();
357 
358                     boolean empty = t == null;
359 
360                     if (checkTerminated(d, empty, a, q)) {
361                         return;
362                     }
363 
364                     if (empty) {
365                         break;
366                     }
367 
368                     a.onNext(t);
369 
370                     r--;
371                     e--;
372                 }
373 
374                 if (e != 0L) {
375                     if (!unbounded) {
376                         REQUESTED.addAndGet(this, e);
377                     }
378                     s.request(-e);
379                 }
380 
381                 missed = WIP.addAndGet(this, -missed);
382                 if (missed == 0) {
383                     break;
384                 }
385             }
386         }
387 
388         void errorAll(Subscriber<? super GroupedObservable<K, V>> a, Queue<?> q, Throwable ex) {
389             q.clear();
390             List<GroupedUnicast<K, V>> list = new ArrayList<GroupedUnicast<K, V>>(groups.values());
391             groups.clear();
392             if (evictedKeys != null) {
393                 evictedKeys.clear();
394             }
395 
396             for (GroupedUnicast<K, V> e : list) {
397                 e.onError(ex);
398             }
399 
400             a.onError(ex);
401         }
402 
403         boolean checkTerminated(boolean d, boolean empty,
404                 Subscriber<? super GroupedObservable<K, V>> a, Queue<?> q) {
405             if (d) {
406                 Throwable err = error;
407                 if (err != null) {
408                     errorAll(a, q, err);
409                     return true;
410                 } else if (empty) {
411                     actual.onCompleted();
412                     return true;
413                 }
414             }
415             return false;
416         }

421         public static <T, K> GroupedUnicast<K, T> createWith(K key, int bufferSize,
422                 GroupBySubscriber<?, K, T> parent, boolean delayError) {
423             State<T, K> state = new State<T, K>(bufferSize, parent, key, delayError);
424             return new GroupedUnicast<K, T>(key, state);
425         }

427         final State<T, K> state;
428 
429         protected GroupedUnicast(K key, State<T, K> state) {
430             super(key, state);
431             this.state = state;
432         }
433 
434         public void onNext(T t) {
435             state.onNext(t);
436         }
437 
438         public void onError(Throwable e) {
439             state.onError(e);
440         }
441 
442         public void onComplete() {
443             state.onComplete();
444         }

447     static final class State<T, K> extends AtomicInteger
448             implements Producer, Subscription, OnSubscribe<T> {
449         /** */
450         private static final long serialVersionUID = -3852313036005250360L;
451 
452         final K key;
453         final Queue<Object> queue;
454         final GroupBySubscriber<?, K, T> parent;
455         final boolean delayError;
456 
457         volatile long requested;
458         @SuppressWarnings("rawtypes")
459         static final AtomicLongFieldUpdater<State> REQUESTED = AtomicLongFieldUpdater
460                 .newUpdater(State.class, "requested");
461 
462         volatile boolean done;
463         Throwable error;
464 
465         volatile int cancelled;
466         @SuppressWarnings("rawtypes")
467         static final AtomicIntegerFieldUpdater<State> CANCELLED = AtomicIntegerFieldUpdater
468                 .newUpdater(State.class, "cancelled");
469 
470         volatile Subscriber<? super T> actual;
471         @SuppressWarnings("rawtypes")
472         static final AtomicReferenceFieldUpdater<State, Subscriber> ACTUAL = AtomicReferenceFieldUpdater
473                 .newUpdater(State.class, Subscriber.class, "actual");
474 
475         volatile int once;
476         @SuppressWarnings("rawtypes")
477         static final AtomicIntegerFieldUpdater<State> ONCE = AtomicIntegerFieldUpdater
478                 .newUpdater(State.class, "once");
479 
480         public State(int bufferSize, GroupBySubscriber<?, K, T> parent, K key, boolean delayError) {
481             this.queue = new ConcurrentLinkedQueue<Object>();
482             this.parent = parent;
483             this.key = key;
484             this.delayError = delayError;
485         }
486 
487         @Override
488         public void request(long n) {
489             if (n < 0) {
490                 throw new IllegalArgumentException("n >= required but it was " + n);
491             }
492             if (n != 0L) {
493                 BackpressureUtils.getAndAddRequest(REQUESTED, this, n);
494                 drain();
495             }
496         }
497 
498         @Override
499         public boolean isUnsubscribed() {
500             return cancelled != 0;
501         }
502 
503         @Override
504         public void unsubscribe() {
505             if (CANCELLED.compareAndSet(this, 0, 1)) {
506                 if (getAndIncrement() == 0) {
507                     parent.cancel(key);
508                 }
509             }
510         }
511 
512         @Override
513         public void call(Subscriber<? super T> s) {
514             if (ONCE.compareAndSet(this, 0, 1)) {
515                 s.add(this);
516                 s.setProducer(this);
517                 ACTUAL.lazySet(this, s);
518                 drain();
519             } else {
520                 s.onError(new IllegalStateException("Only one Subscriber allowed!"));
521             }
522         }
523 
524         public void onNext(T t) {
525             if (t == null) {
526                 error = new NullPointerException();
527                 done = true;
528             } else {
529                 queue.offer(NotificationLite.instance().next(t));
530             }
531             drain();
532         }
533 
534         public void onError(Throwable e) {
535             error = e;
536             done = true;
537             drain();
538         }
539 
540         public void onComplete() {
541             done = true;
542             drain();
543         }
544 
545         void drain() {
546             if (getAndIncrement() != 0) {
547                 return;
548             }
549             int missed = 1;
550 
551             final Queue<Object> q = queue;
552             final boolean delayError = this.delayError;
553             Subscriber<? super T> a = actual;
554             NotificationLite<T> nl = NotificationLite.instance();
555             for (;;) {
556                 if (a != null) {
557                     if (checkTerminated(done, q.isEmpty(), a, delayError)) {
558                         return;
559                     }
560 
561                     long r = requested;
562                     boolean unbounded = r == Long.MAX_VALUE;
563                     long e = 0;
564 
565                     while (r != 0L) {
566                         boolean d = done;
567                         Object v = q.poll();
568                         boolean empty = v == null;
569 
570                         if (checkTerminated(d, empty, a, delayError)) {
571                             return;
572                         }
573 
574                         if (empty) {
575                             break;
576                         }
577 
578                         a.onNext(nl.getValue(v));
579 
580                         r--;
581                         e--;
582                     }
583 
584                     if (e != 0L) {
585                         if (!unbounded) {
586                             REQUESTED.addAndGet(this, e);
587                         }
588                         parent.s.request(-e);
589                     }
590                 }
591 
592                 missed = addAndGet(-missed);
593                 if (missed == 0) {
594                     break;
595                 }
596                 if (a == null) {
597                     a = actual;
598                 }
599             }
600         }
601 
602         boolean checkTerminated(boolean d, boolean empty, Subscriber<? super T> a,
603                 boolean delayError) {
604             if (cancelled != 0) {
605                 queue.clear();
606                 parent.cancel(key);
607                 return true;
608             }
609 
610             if (d) {
611                 if (delayError) {
612                     if (empty) {
613                         Throwable e = error;
614                         if (e != null) {
615                             a.onError(e);
616                         } else {
617                             a.onCompleted();
618                         }
619                         return true;
620                     }
621                 } else {
622                     Throwable e = error;
623                     if (e != null) {
624                         queue.clear();
625                         a.onError(e);
626                         return true;
627                     } else if (empty) {
628                         a.onCompleted();
629                         return true;
630                     }
631                 }
632             }
633 
634             return false;
635         }
