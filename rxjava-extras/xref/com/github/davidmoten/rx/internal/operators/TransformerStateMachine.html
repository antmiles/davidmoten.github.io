16  public final class TransformerStateMachine<State, In, Out> implements Transformer<In, Out> {
17  
18      private final Func0<? extends State> initialState;
19      private final Func3<? super State, ? super In, ? super Subscriber<Out>, ? extends State> transition;
20      private final Func2<? super State, ? super Subscriber<Out>, Boolean> completion;
21      private final BackpressureStrategy backpressureStrategy;
22  
23      private TransformerStateMachine(Func0<? extends State> initialState,
24              Func3<? super State, ? super In, ? super Subscriber<Out>, ? extends State> transition,
25              Func2<? super State, ? super Subscriber<Out>, Boolean> completion,
26              BackpressureStrategy backpressureStrategy) {
27          Preconditions.checkNotNull(initialState);
28          Preconditions.checkNotNull(transition);
29          Preconditions.checkNotNull(completion);
30          Preconditions.checkNotNull(backpressureStrategy);
31          this.initialState = initialState;
32          this.transition = transition;
33          this.completion = completion;
34          this.backpressureStrategy = backpressureStrategy;
35      }

37      public static <State, In, Out> Transformer<In, Out> create(Func0<? extends State> initialState,
38              Func3<? super State, ? super In, ? super Subscriber<Out>, ? extends State> transition,
39              Func2<? super State, ? super Subscriber<Out>, Boolean> completion,
40              BackpressureStrategy backpressureStrategy) {
41          return new TransformerStateMachine<State, In, Out>(initialState, transition, completion,
42                  backpressureStrategy);
43      }
44  
45      @Override
46      public Observable<Out> call(final Observable<In> source) {
47          // use defer so we can have a single state reference for each
48          // subscription
49          return Observable.defer(new Func0<Observable<Out>>() {
50              @Override
51              public Observable<Out> call() {
52                  Mutable<State> state = new Mutable<State>(initialState.call());
53                  return source.materialize()
54                          // do state transitions and emit notifications
55                          // use flatMap to emit notification values
56                          .flatMap(execute(transition, completion, state, backpressureStrategy))
57                          // filter out unsubscribed error notification that is
58                          // generated by unsubscribing in the transition
59                          .materialize()
60                          // don't pass through an unsubscribed exception (its
61                          // just flow control not a real exception)
62                          .takeWhile(NOT_UNSUBSCRIBED)
63                          // flatten notifications of notifications to
64                          // notifications
65                          .dematerialize()
66                          // flatten notifications to a stream which will enable
67                          // early termination from the state machine if desired
68                          .dematerialize();
69              }
70          });
71      }

73      private static <State, Out, In> Func1<Notification<In>, Observable<Notification<Out>>> execute(
74              final Func3<? super State, ? super In, ? super Subscriber<Out>, ? extends State> transition,
75              final Func2<? super State, ? super Subscriber<Out>, Boolean> completion,
76              final Mutable<State> state, final BackpressureStrategy backpressureStrategy) {
77  
78          return new Func1<Notification<In>, Observable<Notification<Out>>>() {
79  
80              @Override
81              public Observable<Notification<Out>> call(final Notification<In> in) {
82  
83                  Observable<Notification<Out>> o = Observable
84                          .create(new OnSubscribe<Notification<Out>>() {
85  
86                      @Override
87                      public void call(Subscriber<? super Notification<Out>> subscriber) {
88                          Subscriber<Out> w = wrap(subscriber);
89                          if (in.hasValue()) {
90                              state.value = transition.call(state.value, in.getValue(), w);
91                              if (!subscriber.isUnsubscribed())
92                                  subscriber.onCompleted();
93                              else {
94                                  subscriber.onError(UnsubscribedExceptionHolder.INSTANCE);
95                              }
96                          } else if (in.isOnCompleted()) {
97                              if (completion.call(state.value, w) && !subscriber.isUnsubscribed()) {
98                                  w.onCompleted();
99                              }
100                         } else if (!subscriber.isUnsubscribed()) {
101                             w.onError(in.getThrowable());
102                         }
103                     }
104                 });
105                 // because the observable we just created does not
106                 // support backpressure we need to apply a backpressure
107                 // handling operator. This operator is supplied by the
108                 // user.
109                 return applyBackpressure(o, backpressureStrategy);
110             }
111 
112         };
113     }
114 
115     private static <Out> Observable<Notification<Out>> applyBackpressure(
116             Observable<Notification<Out>> o, final BackpressureStrategy backpressureStrategy) {
117         if (backpressureStrategy == BackpressureStrategy.BUFFER)
118             return o.onBackpressureBuffer();
119         else if (backpressureStrategy == BackpressureStrategy.DROP)
120             return o.onBackpressureDrop();
121         else if (backpressureStrategy == BackpressureStrategy.LATEST)
122             return o.onBackpressureLatest();
123         else
124             throw new IllegalArgumentException(
125                     "backpressure strategy not supported: " + backpressureStrategy);
126     }

128     private static final Func1<Notification<?>, Boolean> NOT_UNSUBSCRIBED = new Func1<Notification<?>, Boolean>() {
129 
130         @Override
131         public Boolean call(Notification<?> t) {
132             return !t.isOnError() || t.getThrowable() != UnsubscribedExceptionHolder.INSTANCE;
133         }

153     private static <Out> NotificationSubscriber<Out> wrap(

158     private static final class NotificationSubscriber<Out> extends Subscriber<Out> {
159 
160         private final Subscriber<? super Notification<Out>> sub;
161 
162         NotificationSubscriber(Subscriber<? super Notification<Out>> sub) {
163             this.sub = sub;
164             add(sub);
165         }
166 
167         @Override
168         public void onCompleted() {
169             sub.onNext(Notification.<Out> createOnCompleted());
170         }
171 
172         @Override
173         public void onError(Throwable e) {
174             sub.onNext(Notification.<Out> createOnError(e));
175         }
176 
177         @Override
178         public void onNext(Out t) {
179             sub.onNext(Notification.createOnNext(t));
180         }
