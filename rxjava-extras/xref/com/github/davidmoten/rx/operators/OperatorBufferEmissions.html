1   package com.github.davidmoten.rx.operators;
2   
3   import java.util.concurrent.ConcurrentLinkedQueue;
4   import java.util.concurrent.atomic.AtomicReference;
5   
6   import rx.Observable.Operator;
7   import rx.Producer;
8   import rx.Scheduler;
9   import rx.Subscriber;
10  
11  import com.github.davidmoten.rx.util.Drainer;
12  import com.github.davidmoten.rx.util.DrainerAsyncBiased;
13  import com.github.davidmoten.rx.util.DrainerSyncBiased;
14  
15  public class OperatorBufferEmissions<T> implements Operator<T, T> {
16  
17      private final Scheduler observeOnScheduler;
18  
19      public OperatorBufferEmissions() {
20          this(null);
21      }
22  
23      public OperatorBufferEmissions(Scheduler observeOnScheduler) {
24          this.observeOnScheduler = observeOnScheduler;
25      }
26  
27      @Override
28      public Subscriber<? super T> call(Subscriber<? super T> child) {
29          // need to keep an atomic reference to drainer because parent refers to
30          // drainer and drainer refers to parent
31          final AtomicReference<Drainer<T>> drainerRef = new AtomicReference<Drainer<T>>();
32          final ParentSubscriber<T> parent = new ParentSubscriber<T>(drainerRef);
33          Producer requestFromUpstream = new Producer() {
34              @Override
35              public void request(long n) {
36                  parent.requestMore(n);
37              }
38          };
39          final Drainer<T> drainer = createDrainer(child, requestFromUpstream, observeOnScheduler);
40          drainerRef.set(drainer);
41          child.add(parent);
42          child.setProducer(new Producer() {
43              @Override
44              public void request(long n) {
45                  drainer.request(n);
46              }
47          });
48          return parent;
49      }
50  
51      private static <T> Drainer<T> createDrainer(Subscriber<? super T> child,
52              Producer requestFromUpstream, Scheduler observeOnScheduler) {
53          final Drainer<T> drainer;
54          if (observeOnScheduler == null)
55              drainer = DrainerSyncBiased.create(new ConcurrentLinkedQueue<T>(), child,
56                      requestFromUpstream);
57          else
58              drainer = DrainerAsyncBiased.create(new ConcurrentLinkedQueue<Object>(), child,
59                      observeOnScheduler.createWorker(), child, requestFromUpstream);
60          return drainer;
61      }
62  
63      private static class ParentSubscriber<T> extends Subscriber<T> {
64  
65          private final AtomicReference<Drainer<T>> drainerRef;
66  
67          ParentSubscriber(final AtomicReference<Drainer<T>> drainerRef) {
68              this.drainerRef = drainerRef;
69          }
70  
71          void requestMore(long n) {
72              if (n > 0)
73                  request(n);
74          }
75  
76          @Override
77          public void onCompleted() {
78              drainerRef.get().onCompleted();
79          }
80  
81          @Override
82          public void onError(Throwable e) {
83              drainerRef.get().onError(e);
84          }
85  
86          @Override
87          public void onNext(T t) {
88              drainerRef.get().onNext(t);
89          }
90  
91      };
92  
93  }
