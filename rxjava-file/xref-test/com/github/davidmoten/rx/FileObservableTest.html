67      @Test
68      public void testCreateAndModifyEventsForANonDirectoryFile()
69              throws InterruptedException, IOException {
70          File file = new File("target/f");
71          file.delete();
72          Observable<WatchEvent<?>> events = FileObservable.from(file, ENTRY_CREATE, ENTRY_MODIFY);
73          final CountDownLatch latch = new CountDownLatch(1);
74          @SuppressWarnings("unchecked")
75          final List<Kind<?>> eventKinds = Mockito.mock(List.class);
76          InOrder inOrder = Mockito.inOrder(eventKinds);
77          final AtomicInteger errorCount = new AtomicInteger(0);
78          Subscription sub = events.subscribeOn(Schedulers.io())
79                  .subscribe(new Observer<WatchEvent<?>>() {
80  
81                      @Override
82                      public void onCompleted() {
83                          System.out.println("completed");
84                      }
85  
86                      @Override
87                      public void onError(Throwable e) {
88                          errorCount.incrementAndGet();
89                      }
90  
91                      @Override
92                      public void onNext(WatchEvent<?> event) {
93                          System.out.println("event=" + event);
94                          eventKinds.add(event.kind());
95                          latch.countDown();
96                      }
97                  });
98          // sleep long enough for WatchService to start
99          Thread.sleep(1000);
100         file.createNewFile();
101         FileOutputStream fos = new FileOutputStream(file, true);
102         fos.write("hello there".getBytes());
103         fos.close();
104         // give the WatchService time to register the change
105         Thread.sleep(100);
106         assertTrue(latch.await(30000, TimeUnit.MILLISECONDS));
107         inOrder.verify(eventKinds).add(StandardWatchEventKinds.ENTRY_CREATE);
108         inOrder.verify(eventKinds).add(StandardWatchEventKinds.ENTRY_MODIFY);
109         inOrder.verifyNoMoreInteractions();
110 
111         sub.unsubscribe();
112         Thread.sleep(100);
113         assertEquals(0, errorCount.get());
114     }

116     @Test
117     public void testFileTailingFromStartOfFile() throws InterruptedException, IOException {
118         final File log = new File("target/test.log");
119         log.delete();
120         log.createNewFile();
121         append(log, "a0");
122 
123         Observable<String> tailer = FileObservable.tailer().file(log).onWatchStarted(new Action0() {
124             @Override
125             public void call() {
126                 append(log, "a1");
127                 append(log, "a2");
128             }
129         }).sampleTimeMs(50).utf8().tailText();
130         final List<String> list = new ArrayList<String>();
131         final CountDownLatch latch = new CountDownLatch(3);
132         Subscription sub = tailer.subscribeOn(Schedulers.io()).subscribe(new Action1<String>() {
133             @Override
134             public void call(String line) {
135                 System.out.println("received: '" + line + "'");
136                 list.add(line);
137                 latch.countDown();
138             }
139         });
140         assertTrue(latch.await(10, TimeUnit.SECONDS));
141         assertEquals(Arrays.asList("a0", "a1", "a2"), list);
142         sub.unsubscribe();
143     }

181     private static void append(File file, String line) {
182         try {

186             fos.close();

194     @Test
195     public void testTailTextFileStreamsFromEndOfFileIfSpecified()
196             throws FileNotFoundException, InterruptedException {
197         File file = new File("target/test1.txt");
198         file.delete();
199         try (PrintStream out = new PrintStream(file)) {
200             out.println("line 1");
201         }
202         final List<String> list = new ArrayList<String>();
203         TestSubscriber<String> ts = TestSubscriber.create();
204         FileObservable.tailer().file(file).startPosition(file.length()).sampleTimeMs(10).utf8()
205                 .tailText()
206                 // for each
207                 .doOnNext(new Action1<String>() {
208 
209                     @Override
210                     public void call(String line) {
211                         System.out.println(line);
212                         list.add(line);
213                     }
214                 }).subscribeOn(Schedulers.newThread()).subscribe(ts);
215         Thread.sleep(1100);
216         assertTrue(list.isEmpty());
217         try (PrintStream out = new PrintStream(new FileOutputStream(file, true))) {
218             out.println("line 2");
219         }
220         Thread.sleep(1100);
221         assertEquals(1, list.size());
222         assertEquals("line 2", list.get(0).trim());
223         ts.unsubscribe();
224     }

228             throws InterruptedException, IOException {
229         File file = new File("target/test2.txt");
230         file.delete();
231         try (PrintStream out = new PrintStream(file)) {
232             out.println("line 1");
233         }
234         final List<String> list = new ArrayList<String>();
235         Subscription sub = FileObservable.tailer().file(file).startPosition(file.length())
236                 .sampleTimeMs(10).utf8().tailText()
237                 // for each
238                 .doOnNext(new Action1<String>() {
239 
240                     @Override
241                     public void call(String line) {
242                         System.out.println(line);
243                         list.add(line);
244                     }
245                 }).subscribeOn(Schedulers.newThread()).subscribe();
246         // delay must be long enough for last update timestamp to change on
247         // windows (resolution to the second)
248         Thread.sleep(1100);
249         assertTrue(list.isEmpty());
250         // delete file then make it bigger than it was
251         assertTrue(file.delete());
252         try (PrintStream out = new PrintStream(new FileOutputStream(file, true))) {
253             out.println("line 2");
254             out.println("line 3");
255         }
256         Thread.sleep(1100);
257         assertEquals(2, list.size());
258         assertEquals("line 2", list.get(0).trim());
259         assertEquals("line 3", list.get(1).trim());
260         sub.unsubscribe();
