21  /**
22   * Observable utility methods related to {@link File}.
23   */
24  public final class FileObservable {
25  
26      /**
27       * Returns an {@link Observable} that uses NIO WatchService (and a dedicated
28       * thread) to push modify events to an observable that reads and reports new
29       * lines to a subscriber. The NIO WatchService events are sampled according
30       * to <code>sampleTimeMs</code> so that lots of discrete activity on a file
31       * (for example a log file with very frequent entries) does not prompt an
32       * inordinate number of file reads to pick up changes.
33       * 
34       * @param file
35       *            the file to tail
36       * @param startPosition
37       *            start tailing file at position in bytes
38       * @param sampleTimeMs
39       *            sample time in millis
40       * @return
41       */
42      public final static Observable<String> tailFile(File file, long startPosition, long sampleTimeMs) {
43          return from(file, StandardWatchEventKinds.ENTRY_CREATE, StandardWatchEventKinds.ENTRY_MODIFY,
44                  StandardWatchEventKinds.OVERFLOW)
45          // don't care about the event details, just that there is one
46                  .cast(Object.class)
47                  // get lines once on subscription so we tail the lines in the
48                  // file at startup
49                  .startWith(new Object())
50                  // emit a max of 1 event per sample period
51                  .sample(sampleTimeMs, TimeUnit.MILLISECONDS)
52                  // tail file triggered by events
53                  .lift(new OperatorFileTailer(file, startPosition));
54      }
55  
56      /**
57       * Returns an {@link Observable} of {@link WatchEvent}s from a
58       * {@link WatchService}.
59       * 
60       * @param watchService
61       *            {@link WatchService} to generate events for
62       * @return
63       */
64      public final static Observable<WatchEvent<?>> from(WatchService watchService) {
65          return Observable.from(watchService).lift(new OperatorWatchServiceEvents());
66      }
67  
68      /**
69       * If file does not exist at subscribe time then is assumed to not be a
70       * directory. If the file is not a directory (bearing in mind the aforesaid
71       * assumption) then a {@link WatchService} is set up on its parent and
72       * {@link WatchEvent}s of the given kinds are filtered to concern the file
73       * in question. If the file is a directory then a {@link WatchService} is
74       * set up on the directory and all events are passed through of the given
75       * kinds.
76       * 
77       * @param file
78       *            file to watch
79       * @param kinds
80       *            event kinds to watch for and emit
81       * @return
82       */
83      @SafeVarargs
84      public final static Observable<WatchEvent<?>> from(final File file, Kind<?>... kinds) {
85          return watchService(file, kinds).flatMap(TO_WATCH_EVENTS).filter(onlyRelatedTo(file));
86      }
87  
88      /**
89       * Creates a {@link WatchService} on subscribe for the given file and event
90       * kinds.
91       * 
92       * @param file
93       *            the file to watch
94       * @param kinds
95       *            event kinds to watch for
96       * @return
97       */
98      @SafeVarargs
99      public final static Observable<WatchService> watchService(final File file, final Kind<?>... kinds) {
100         return Observable.create(new OnSubscribe<WatchService>() {
101 
102             @Override
103             public void call(Subscriber<? super WatchService> subscriber) {
104                 final Path path = getBasePath(file);
105                 try {
106                     WatchService watchService = path.getFileSystem().newWatchService();
107                     path.register(watchService, kinds);
108                     subscriber.onNext(watchService);
109                     subscriber.onCompleted();
110                 } catch (IOException e) {
111                     subscriber.onError(e);
112                 }
113             }
114 
115         });
116     }
117 
118     private final static Path getBasePath(final File file) {
119         final Path path;
120         if (file.exists() && file.isDirectory())
121             path = Paths.get(file.toURI());
122         else
123             path = Paths.get(file.getParentFile().toURI());
124         return path;
125     }
126 
127     /**
128      * Returns true if and only if the path corresponding to a WatchEvent
129      * represents the given file. This will be the case for Create, Modify,
130      * Delete events.
131      * 
132      * @param file
133      *            the file to restrict events to
134      * @return
135      */
136     private final static Func1<WatchEvent<?>, Boolean> onlyRelatedTo(final File file) {
137         return new Func1<WatchEvent<?>, Boolean>() {
138 
139             @Override
140             public Boolean call(WatchEvent<?> event) {
141 
142                 final boolean ok;
143                 if (file.isDirectory())
144                     ok = true;
145                 else if (StandardWatchEventKinds.OVERFLOW.equals(event.kind()))
146                     // TODO allow overflow events through?
147                     ok = true;
148                 else {
149                     Object context = event.context();
150                     if (context != null && context instanceof Path) {
151                         Path p = (Path) context;
152                         Path basePath = getBasePath(file);
153                         File pFile = new File(basePath.toFile(), p.toString());
154                         ok = pFile.getAbsolutePath().equals(file.getAbsolutePath());
155                     } else
156                         ok = false;
157                 }
158                 return ok;
159             }
160         };
161     }
162 
163     private final static Func1<WatchService, Observable<WatchEvent<?>>> TO_WATCH_EVENTS = new Func1<WatchService, Observable<WatchEvent<?>>>() {
164 
165         @Override
166         public Observable<WatchEvent<?>> call(WatchService watchService) {
167             return from(watchService);
168         }
169     };
