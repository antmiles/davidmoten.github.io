29      /**
30       * Returns an {@link Observable} that uses NIO WatchService (and a dedicated
31       * thread) to push modify events to an observable that reads and reports new
32       * lines to a subscriber. The NIO WatchService events are sampled according
33       * to <code>sampleTimeMs</code> so that lots of discrete activity on a file
34       * (for example a log file with very frequent entries) does not prompt an
35       * inordinate number of file reads to pick up changes.
36       * 
37       * @param file
38       *            the file to tail
39       * @param startPosition
40       *            start tailing file at position in bytes
41       * @param sampleTimeMs
42       *            sample time in millis
43       * @return
44       */
45      public final static Observable<String> tailFile(File file, long startPosition, long sampleTimeMs) {
46          return from(file, StandardWatchEventKinds.ENTRY_CREATE, StandardWatchEventKinds.ENTRY_MODIFY,
47                  StandardWatchEventKinds.OVERFLOW)
48          // don't care about the event details, just that there is one
49                  .cast(Object.class)
50                  // get lines once on subscription so we tail the lines in the
51                  // file at startup
52                  .startWith(new Object())
53                  // emit a max of 1 event per sample period
54                  .sample(sampleTimeMs, TimeUnit.MILLISECONDS)
55                  // tail file triggered by events
56                  .lift(new OperatorFileTailer(file, startPosition));
57      }

59      /**
60       * Returns an {@link Observable} of {@link WatchEvent}s from a
61       * {@link WatchService}.
62       * 
63       * @param watchService
64       *            {@link WatchService} to generate events for
65       * @return
66       */
67      public final static Observable<WatchEvent<?>> from(WatchService watchService) {
68          return Observable.create(new WatchServiceOnSubscribe(watchService));
69      }

71      /**
72       * If file does not exist at subscribe time then is assumed to not be a
73       * directory. If the file is not a directory (bearing in mind the aforesaid
74       * assumption) then a {@link WatchService} is set up on its parent and
75       * {@link WatchEvent}s of the given kinds are filtered to concern the file
76       * in question. If the file is a directory then a {@link WatchService} is
77       * set up on the directory and all events are passed through of the given
78       * kinds.
79       * 
80       * @param file
81       *            file to watch
82       * @param kinds
83       *            event kinds to watch for and emit
84       * @return
85       */
86      @SafeVarargs
87      public final static Observable<WatchEvent<?>> from(final File file, Kind<?>... kinds) {
88          return watchService(file, kinds).flatMap(TO_WATCH_EVENTS).filter(onlyRelatedTo(file));
89      }

91      /**
92       * Creates a {@link WatchService} on subscribe for the given file and event
93       * kinds.
94       * 
95       * @param file
96       *            the file to watch
97       * @param kinds
98       *            event kinds to watch for
99       * @return
100      */
101     @SafeVarargs
102     public final static Observable<WatchService> watchService(final File file, final Kind<?>... kinds) {
103         return Observable.create(new OnSubscribe<WatchService>() {
104 
105             @Override
106             public void call(Subscriber<? super WatchService> subscriber) {
107                 final Path path = getBasePath(file);
108                 try {
109                     WatchService watchService = path.getFileSystem().newWatchService();
110                     path.register(watchService, kinds);
111                     subscriber.onNext(watchService);
112                     subscriber.onCompleted();
113                 } catch (IOException e) {
114                     subscriber.onError(e);
115                 }
116             }
117 
118         });
119     }
120 
121     private final static Path getBasePath(final File file) {
122         final Path path;
123         if (file.exists() && file.isDirectory())
124             path = Paths.get(file.toURI());
125         else
126             path = Paths.get(file.getParentFile().toURI());
127         return path;
128     }
129 
130     /**
131      * Returns true if and only if the path corresponding to a WatchEvent
132      * represents the given file. This will be the case for Create, Modify,
133      * Delete events.
134      * 
135      * @param file
136      *            the file to restrict events to
137      * @return
138      */
139     private final static Func1<WatchEvent<?>, Boolean> onlyRelatedTo(final File file) {
140         return new Func1<WatchEvent<?>, Boolean>() {
141 
142             @Override
143             public Boolean call(WatchEvent<?> event) {
144 
145                 final boolean ok;
146                 if (file.isDirectory())
147                     ok = true;
148                 else if (StandardWatchEventKinds.OVERFLOW.equals(event.kind()))
149                     // TODO allow overflow events through?
150                     ok = true;
151                 else {
152                     Object context = event.context();
153                     if (context != null && context instanceof Path) {
154                         Path p = (Path) context;
155                         Path basePath = getBasePath(file);
156                         File pFile = new File(basePath.toFile(), p.toString());
157                         ok = pFile.getAbsolutePath().equals(file.getAbsolutePath());
158                     } else
159                         ok = false;
160                 }
161                 return ok;
162             }
163         };
164     }

166     static class WatchServiceOnSubscribe implements OnSubscribe<WatchEvent<?>> {
167 
168         private final WatchService watchService;
169         private final AtomicBoolean subscribed = new AtomicBoolean(true);
170 
171         WatchServiceOnSubscribe(WatchService watchService) {
172             this.watchService = watchService;
173         }
174 
175         @Override
176         public void call(Subscriber<? super WatchEvent<?>> subscriber) {
177 
178             if (!subscribed.get()) {
179                 subscriber.onError(new RuntimeException(
180                         "WatchService closed. You can only subscribe once to a WatchService."));
181                 return;
182             }
183             subscriber.add(createSubscriptionToCloseWatchService(watchService, subscribed, subscriber));
184 
185             emitEvents(subscriber);
186         }
187 
188         private void emitEvents(Subscriber<? super WatchEvent<?>> subscriber) {
189             // get the first event before looping
190             WatchKey key = nextKey(subscriber);
191 
192             while (key != null) {
193                 if (subscriber.isUnsubscribed())
194                     return;
195                 // we have a polled event, now we traverse it and
196                 // receive all the states from it
197                 for (WatchEvent<?> event : key.pollEvents()) {
198                     if (subscriber.isUnsubscribed())
199                         return;
200                     else
201                         subscriber.onNext(event);
202                 }
203 
204                 boolean valid = key.reset();
205                 if (!valid && subscribed.get()) {
206                     subscriber.onCompleted();
207                     return;
208                 } else if (!valid)
209                     return;
210 
211                 key = nextKey(subscriber);
212             }
213         }
214 
215         private WatchKey nextKey(Subscriber<? super WatchEvent<?>> subscriber) {
216             try {
217                 // this command blocks but unsubscribe close the watch
218                 // service and interrupt it
219                 return watchService.take();
220             } catch (ClosedWatchServiceException e) {
221                 // must have unsubscribed
222                 if (subscribed.get())
223                     subscriber.onCompleted();
224                 return null;
225             } catch (InterruptedException e) {
226                 // this case is problematic because unsubscribe may call
227                 // Thread.interrupt() before calling the unsubscribe method of
228                 // the Subscription. Thus at this point we don't know if a
229                 // deliberate interrupt was called in which case I would call
230                 // onComplete or if unsubscribe was called in which case I
231                 // should not call anything. For the moment I choose to not call
232                 // anything partly because a deliberate stop of the
233                 // watchService.take ignorant of the Observable should ideally
234                 // happen via a call to the WatchService.close() method rather
235                 // than Thread.interrupt().
236                 // TODO raise the issue with RxJava team in particular
237                 // Subscriptions.from(Future) calling FutureTask.cancel(true)
238                 try {
239                     watchService.close();
240                 } catch (IOException e1) {
241                     // do nothing
242                 }
243                 return null;
244             }
245         }
246 
247     }

249     private final static Subscription createSubscriptionToCloseWatchService(final WatchService watchService,
250             final AtomicBoolean subscribed, final Subscriber<? super WatchEvent<?>> subscriber) {
251         return new Subscription() {
252 
253             @Override
254             public void unsubscribe() {
255                 try {
256                     watchService.close();
257                     subscribed.set(false);
258                 } catch (ClosedWatchServiceException e) {
259                     // do nothing
260                     subscribed.set(false);
261                 } catch (IOException e) {
262                     // do nothing
263                     subscribed.set(false);
264                 }
265             }
266 
267             @Override
268             public boolean isUnsubscribed() {
269                 return !subscribed.get();
270             }
271         };
272     }

277         public Observable<WatchEvent<?>> call(WatchService watchService) {
278             return from(watchService);
279         }
