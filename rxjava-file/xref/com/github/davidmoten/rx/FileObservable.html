28      public static final int DEFAULT_MAX_BYTES_PER_EMISSION = 8192;

30      /**
31       * Returns an {@link Observable} that uses NIO {@link WatchService} (and a
32       * dedicated thread) to push modified events to an observable that reads and
33       * reports new sequences of bytes to a subscriber. The NIO
34       * {@link WatchService} events are sampled according to
35       * <code>sampleTimeMs</code> so that lots of discrete activity on a file
36       * (for example a log file with very frequent entries) does not prompt an
37       * inordinate number of file reads to pick up changes.
38       * 
39       * @param file
40       *            the file to tail
41       * @param startPosition
42       *            start tailing file at position in bytes
43       * @param sampleTimeMs
44       *            sample time in millis
45       * @param chunkSize
46       *            max array size of each element emitted by the Observable. Is
47       *            also used as the buffer size for reading from the file. Try
48       *            {@link FileObservable#DEFAULT_MAX_BYTES_PER_EMISSION} if you
49       *            don't know what to put here.
50       * @return
51       */
52      public final static Observable<byte[]> tailFile(File file, long startPosition, long sampleTimeMs, int chunkSize) {
53          Observable<Object> events = from(file, StandardWatchEventKinds.ENTRY_CREATE,
54                  StandardWatchEventKinds.ENTRY_MODIFY, StandardWatchEventKinds.OVERFLOW)
55          // don't care about the event details, just that there is one
56                  .cast(Object.class)
57                  // get lines once on subscription so we tail the lines
58                  // in the file at startup
59                  .startWith(new Object());
60          return tailFile(file, startPosition, sampleTimeMs, chunkSize, events);
61      }

63      /**
64       * Returns an {@link Observable} that uses given given observable to push
65       * modified events to an observable that reads and reports new sequences of
66       * bytes to a subscriber. The NIO {@link WatchService} MODIFY and OVERFLOW
67       * events are sampled according to <code>sampleTimeMs</code> so that lots of
68       * discrete activity on a file (for example a log file with very frequent
69       * entries) does not prompt an inordinate number of file reads to pick up
70       * changes. File create events are not sampled and are always passed
71       * through.
72       * 
73       * @param file
74       *            the file to tail
75       * @param startPosition
76       *            start tailing file at position in bytes
77       * @param sampleTimeMs
78       *            sample time in millis for MODIFY and OVERFLOW events
79       * @param chunkSize
80       *            max array size of each element emitted by the Observable. Is
81       *            also used as the buffer size for reading from the file. Try
82       *            {@link FileObservable#DEFAULT_MAX_BYTES_PER_EMISSION} if you
83       *            don't know what to put here.
84       * @return
85       */
86      public final static Observable<byte[]> tailFile(File file, long startPosition, long sampleTimeMs, int chunkSize,
87              Observable<?> events) {
88          return sampleModifyOrOverflowEventsOnly(events, sampleTimeMs)
89          // tail file triggered by events
90                  .lift(new OperatorFileTailer(file, startPosition, chunkSize));
91      }

93      /**
94       * Returns an {@link Observable} that uses NIO {@link WatchService} (and a
95       * dedicated thread) to push modified events to an observable that reads and
96       * reports new lines to a subscriber. The NIO WatchService MODIFY and
97       * OVERFLOW events are sampled according to <code>sampleTimeMs</code> so
98       * that lots of discrete activity on a file (for example a log file with
99       * very frequent entries) does not prompt an inordinate number of file reads
100      * to pick up changes. File create events are not sampled and are always
101      * passed through.
102      * 
103      * @param file
104      *            the file to tail
105      * @param startPosition
106      *            start tailing file at position in bytes
107      * @param sampleTimeMs
108      *            sample time in millis for MODIFY and OVERFLOW events
109      * @param charset
110      *            the character set to use to decode the bytes to a string
111      * @return
112      */
113     public final static Observable<String> tailTextFile(File file, long startPosition, long sampleTimeMs,
114             Charset charset) {
115         return toLines(tailFile(file, startPosition, sampleTimeMs, DEFAULT_MAX_BYTES_PER_EMISSION), charset);
116     }

118     /**
119      * Returns an {@link Observable} of String that uses the given events stream
120      * to trigger checks on file change so that new lines can be read and
121      * emitted.
122      * 
123      * @param file
124      *            the file to tail, cannot be null
125      * @param startPosition
126      *            start tailing file at position in bytes
127      * @param charset
128      *            the character set to use to decode the bytes to a string
129      * @param events
130      *            trigger a check for file changes. Use
131      *            {@link Observable#interval(long, TimeUnit)} for example.
132      * @return
133      */
134     public final static Observable<String> tailTextFile(File file, long startPosition, int chunkSize, Charset charset,
135             Observable<?> events) {
136         return toLines(events.lift(new OperatorFileTailer(file, startPosition, chunkSize)), charset);
137     }

139     /**
140      * Returns an {@link Observable} of {@link WatchEvent}s from a
141      * {@link WatchService}.
142      * 
143      * @param watchService
144      *            {@link WatchService} to generate events for
145      * @return
146      */
147     public final static Observable<WatchEvent<?>> from(WatchService watchService) {
148         return Observable.from(watchService).lift(new OperatorWatchServiceEvents());
149     }

151     /**
152      * If file does not exist at subscribe time then is assumed to not be a
153      * directory. If the file is not a directory (bearing in mind the aforesaid
154      * assumption) then a {@link WatchService} is set up on its parent and
155      * {@link WatchEvent}s of the given kinds are filtered to concern the file
156      * in question. If the file is a directory then a {@link WatchService} is
157      * set up on the directory and all events are passed through of the given
158      * kinds.
159      * 
160      * @param file
161      *            file to watch
162      * @param kinds
163      *            event kinds to watch for and emit
164      * @return
165      */
166     @SafeVarargs
167     public final static Observable<WatchEvent<?>> from(final File file, Kind<?>... kinds) {
168         return watchService(file, kinds)
169         // emit events from the WatchService
170                 .flatMap(TO_WATCH_EVENTS)
171                 // restrict to events related to the file
172                 .filter(onlyRelatedTo(file));
173     }
174 
175     /**
176      * Creates a {@link WatchService} on subscribe for the given file and event
177      * kinds.
178      * 
179      * @param file
180      *            the file to watch
181      * @param kinds
182      *            event kinds to watch for
183      * @return
184      */
185     @SafeVarargs
186     public final static Observable<WatchService> watchService(final File file, final Kind<?>... kinds) {
187         return Observable.create(new OnSubscribe<WatchService>() {
188 
189             @Override
190             public void call(Subscriber<? super WatchService> subscriber) {
191                 try {
192                     final Path path = getBasePath(file);
193                     WatchService watchService = path.getFileSystem().newWatchService();
194                     path.register(watchService, kinds);
195                     subscriber.onNext(watchService);
196                     subscriber.onCompleted();
197                 } catch (Exception e) {
198                     subscriber.onError(e);
199                 }
200             }
201 
202         });
203     }
204 
205     private final static Path getBasePath(final File file) {
206         final Path path;
207         if (file.exists() && file.isDirectory())
208             path = Paths.get(file.toURI());
209         else
210             path = Paths.get(file.getParentFile().toURI());
211         return path;
212     }

214     /**
215      * Returns true if and only if the path corresponding to a WatchEvent
216      * represents the given file. This will be the case for Create, Modify,
217      * Delete events.
218      * 
219      * @param file
220      *            the file to restrict events to
221      * @return
222      */
223     private final static Func1<WatchEvent<?>, Boolean> onlyRelatedTo(final File file) {
224         return new Func1<WatchEvent<?>, Boolean>() {
225 
226             @Override
227             public Boolean call(WatchEvent<?> event) {
228 
229                 final boolean ok;
230                 if (file.isDirectory())
231                     ok = true;
232                 else if (StandardWatchEventKinds.OVERFLOW.equals(event.kind()))
233                     ok = true;
234                 else {
235                     Object context = event.context();
236                     if (context != null && context instanceof Path) {
237                         Path p = (Path) context;
238                         Path basePath = getBasePath(file);
239                         File pFile = new File(basePath.toFile(), p.toString());
240                         ok = pFile.getAbsolutePath().equals(file.getAbsolutePath());
241                     } else
242                         ok = false;
243                 }
244                 return ok;
245             }
246         };
247     }

253     private final static Func1<WatchService, Observable<WatchEvent<?>>> TO_WATCH_EVENTS = new Func1<WatchService, Observable<WatchEvent<?>>>() {
254 
255         @Override
256         public Observable<WatchEvent<?>> call(WatchService watchService) {
257             return from(watchService);
258         }

261     private static Observable<Object> sampleModifyOrOverflowEventsOnly(Observable<?> events, final long sampleTimeMs) {
262         return events
263         // group by true if is modify or overflow, false otherwise
264                 .groupBy(IS_MODIFY_OR_OVERFLOW)
265                 // only sample if is modify or overflow
266                 .flatMap(sampleIfTrue(sampleTimeMs));
267     }
268 
269     private static Func1<GroupedObservable<Boolean, ?>, Observable<?>> sampleIfTrue(final long sampleTimeMs) {
270         return new Func1<GroupedObservable<Boolean, ?>, Observable<?>>() {
271 
272             @Override
273             public Observable<?> call(GroupedObservable<Boolean, ?> group) {
274                 // if is modify or overflow WatchEvent
275                 if (group.getKey())
276                     return group.sample(sampleTimeMs, TimeUnit.MILLISECONDS);
277                 else
278                     return group;
279             }
280         };
281     }
282 
283     private static Func1<Object, Boolean> IS_MODIFY_OR_OVERFLOW = new Func1<Object, Boolean>() {
284 
285         @Override
286         public Boolean call(Object event) {
287             if (event instanceof WatchEvent) {
288                 WatchEvent<?> w = (WatchEvent<?>) event;
289                 String kind = w.kind().name();
290                 if (kind.equals(StandardWatchEventKinds.ENTRY_MODIFY.name())
291                         || kind.equals(StandardWatchEventKinds.OVERFLOW.name())) {
292                     return true;
293                 } else
294                     return false;
295             } else
296                 return false;
297         }
298     };

304     public static class Builder {
305 
306         private File file = null;
307         private long startPosition = 0;
308         private long sampleTimeMs = 500;
309         private int chunkSize = 8192;
310         private Charset charset = Charset.defaultCharset();
311         private Observable<?> source = null;
312 
313         private Builder() {
314         }
315 
316         /**
317          * The file to tail.
318          * 
319          * @param file
320          * @return this
321          */
322         public Builder file(File file) {
323             this.file = file;
324             this.source = from(file, StandardWatchEventKinds.ENTRY_CREATE, StandardWatchEventKinds.ENTRY_MODIFY,
325                     StandardWatchEventKinds.OVERFLOW);
326             return this;
327         }
328 
329         public Builder file(String filename) {
330             return file(new File(filename));
331         }
332 
333         /**
334          * The startPosition in bytes in the file to commence the tail from. 0 =
335          * start of file. Defaults to 0.
336          * 
337          * @param startPosition
338          * @return this
339          */
340         public Builder startPosition(long startPosition) {
341             this.startPosition = startPosition;
342             return this;
343         }
344 
345         /**
346          * Specifies sampling to apply to the source observable (which could be
347          * very busy if a lot of writes are occurring for example). Sampling is
348          * only applied to file updates (MODIFY and OVERFLOW), file creation
349          * events are always passed through. File deletion events are ignored
350          * (in fact are not requested of NIO).
351          * 
352          * @param sampleTimeMs
353          * @return this
354          */
355         public Builder sampleTimeMs(long sampleTimeMs) {
356             this.sampleTimeMs = sampleTimeMs;
357             return this;
358         }
359 
360         /**
361          * Emissions from the tailed file will be no bigger than this.
362          * 
363          * @param chunkSize
364          * @return this
365          */
366         public Builder chunkSize(int chunkSize) {
367             this.chunkSize = chunkSize;
368             return this;
369         }
370 
371         /**
372          * The charset of the file. Only used for tailing a text file.
373          * 
374          * @param charset
375          * @return this
376          */
377         public Builder charset(Charset charset) {
378             this.charset = charset;
379             return this;
380         }
381 
382         /**
383          * The charset of the file. Only used for tailing a text file.
384          * 
385          * @param charset
386          * @return this
387          */
388         public Builder charset(String charset) {
389             return charset(Charset.forName(charset));
390         }
391 
392         public Builder utf8() {
393             return charset("UTF-8");
394         }
395 
396         public Builder source(Observable<?> source) {
397             this.source = source;
398             return this;
399         }
400 
401         public Observable<byte[]> tail() {
402             return tailFile(file, startPosition, sampleTimeMs, chunkSize, source);
403         }
404 
405         public Observable<String> tailText() {
406             return tailTextFile(file, startPosition, chunkSize, charset, source);
407         }
