3   import java.sql.Connection;
4   import java.sql.SQLException;
5   import java.util.concurrent.atomic.AtomicInteger;
6   
7   public class DatabaseCreator {
8   
9       private static AtomicInteger dbNumber = new AtomicInteger();
10  
11      public static String nextUrl() {
12          return "jdbc:h2:mem:test" + dbNumber.incrementAndGet() + ";DB_CLOSE_DELAY=-1";
13      }
14  
15      public static ConnectionProvider connectionProvider() {
16          return new ConnectionProviderFromUrl(nextUrl());
17      }
18  
19      public static Database db() {
20          ConnectionProvider cp = connectionProvider();
21          createDatabase(cp.get());
22          return new Database(cp);
23      }
24  
25      public static void createDatabase(Connection c) {
26          try {
27              c.setAutoCommit(true);
28              c.prepareStatement(
29                      "create table person (name varchar(50) primary key, score int not null,dob date, registered timestamp)")
30                      .execute();
31              c.prepareStatement("insert into person(name,score) values('FRED',21)").execute();
32              c.prepareStatement("insert into person(name,score) values('JOSEPH',34)").execute();
33              c.prepareStatement("insert into person(name,score) values('MARMADUKE',25)").execute();
34  
35              c.prepareStatement("create table person_clob (name varchar(50) not null,  document clob not null)")
36                      .execute();
37              c.prepareStatement("create table person_blob (name varchar(50) not null, document blob not null)")
38                      .execute();
39          } catch (SQLException e) {
40              throw new RuntimeException(e);
41          }
42      }
43  
44  }
