114     public void testTransactionUsingCount() {
115         Database db = db();
116         Func1<? super Integer, Boolean> isZero = new Func1<Integer, Boolean>() {
117             @Override
118             public Boolean call(Integer t1) {
119                 return t1 == 0;
120             }
121         };
122         Observable<Integer> existingRows = db
123         // select names
124                 .select("select name from person where name=?")
125                 // set name parameter
126                 .parameter("FRED")
127                 // is part of transaction
128                 .dependsOn(db.beginTransaction())
129                 // get result count
130                 .count()
131                 // return empty if count = 0
132                 .filter(isZero);
133         Observable<Integer> update = db
134         // insert record if does not exist
135                 .update("insert into person(name,score) values(?,0)")
136                 // get parameters from last query
137                 .parameters(existingRows.map(constant("FRED")))
138                 // return num rows affected
139                 .count();
140 
141         boolean committed = db.commit(update).toBlockingObservable().single();
142         assertTrue(committed);
143     }

514     @Test
515     public void testAutoMap() {
516         TimeZone current = TimeZone.getDefault();
517         try {
518             TimeZone.setDefault(TimeZone.getTimeZone("AEST"));
519             Database db = db();
520             Date dob = new Date(100);
521             long now = System.currentTimeMillis();
522             java.sql.Timestamp registered = new java.sql.Timestamp(now);
523             Observable<Integer> u = db.update("update person set dob=?, registered=? where name=?").parameter(dob)
524                     .parameter(registered).parameter("FRED").count();
525             Person person = db.select("select name,score,dob,registered from person order by name").dependsOn(u)
526                     .autoMap(Person.class).first().toBlockingObservable().single();
527             assertEquals("FRED", person.getName());
528             assertEquals(21, person.getScore(), 0.001);
529             // Dates are truncated to start of day
530             assertEquals(0, (long) person.getDateOfBirth());
531             assertEquals(now, (long) person.getRegistered());
532         } finally {
533             TimeZone.setDefault(current);
534         }
535     }

1255     private static class CountDownConnectionProvider implements ConnectionProvider {
1256         private final ConnectionProvider cp;
1257         private final CountDownLatch closesLatch;
1258         private final CountDownLatch getsLatch;
1259 
1260         CountDownConnectionProvider(int expectedGets, int expectedCloses) {
1261             this.cp = connectionProvider();
1262             DatabaseCreator.createDatabase(cp.get());
1263             this.closesLatch = new CountDownLatch(expectedCloses);
1264             this.getsLatch = new CountDownLatch(expectedGets);
1265         }
1266 
1267         CountDownLatch closesLatch() {
1268             return closesLatch;
1269         }
1270 
1271         CountDownLatch getsLatch() {
1272             return getsLatch;
1273         }
1274 
1275         @Override
1276         public Connection get() {
1277             getsLatch.countDown();
1278             Connection inner = cp.get();
1279             return new CountingConnection(inner, closesLatch);
1280         }
1281 
1282         @Override
1283         public void close() {
1284             cp.close();
1285         }
1286     }
