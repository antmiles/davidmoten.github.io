117         Func1<? super Integer, Boolean> isZero = new Func1<Integer, Boolean>() {

119             public Boolean call(Integer t1) {

121             }
122         };
123         Observable<Integer> existingRows = db
124         // select names
125                 .select("select name from person where name=?")
126                 // set name parameter
127                 .parameter("FRED")
128                 // is part of transaction
129                 .dependsOn(db.beginTransaction())
130                 // get result count
131                 .count()
132                 // return empty if count = 0
133                 .filter(isZero);
134         Observable<Integer> update = db
135         // insert record if does not exist
136                 .update("insert into person(name,score) values(?,0)")
137                 // get parameters from last query
138                 .parameters(existingRows.map(constant("FRED")))
139                 // return num rows affected
140                 .count();
141 
142         boolean committed = db.commit(update).toBlocking().single();
143         assertTrue(committed);

570         try {
571             TimeZone.setDefault(TimeZone.getTimeZone("AEST"));
572             Database db = db();
573             Date dob = new Date(100);
574             long now = System.currentTimeMillis();
575             java.sql.Timestamp registered = new java.sql.Timestamp(now);
576             Observable<Integer> u = db.update("update person set dob=?, registered=? where name=?")
577                     .parameter(dob).parameter(registered).parameter("FRED").count();
578             Person person = db.select("select name,score,dob,registered from person order by name")
579                     .dependsOn(u).autoMap(Person.class).first().toBlocking().single();
580             assertEquals("FRED", person.getName());
581             assertEquals(21, person.getScore(), 0.001);
582             // Dates are truncated to start of day
583             assertEquals(0, (long) person.getDateOfBirth());
584             assertEquals(now, (long) person.getRegistered());
585         } finally {
586             TimeZone.setDefault(current);
587         }

1323                 .lift(db.update("update person set score = ?")
