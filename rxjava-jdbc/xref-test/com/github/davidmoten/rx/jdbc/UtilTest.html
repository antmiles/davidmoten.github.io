101     @Test
102     public void testAutoMapBlobToByteArray() throws SQLException {
103         Blob blob = EasyMock.createMock(Blob.class);
104         byte[] b = "hello there".getBytes();
105         expect(blob.getBinaryStream()).andReturn(new ByteArrayInputStream(b)).once();
106         blob.free();
107         EasyMock.expectLastCall().once();
108         replay(blob);
109         Object bytes = autoMap(blob, byte[].class);
110         Arrays.equals(b, (byte[]) bytes);
111         verify(blob);
112     }
