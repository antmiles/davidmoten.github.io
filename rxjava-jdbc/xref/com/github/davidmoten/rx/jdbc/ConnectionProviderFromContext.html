1   package com.github.davidmoten.rx.jdbc;
2   
3   import java.sql.Connection;
4   import java.sql.SQLException;
5   
6   import javax.naming.Context;
7   import javax.naming.InitialContext;
8   import javax.naming.NamingException;
9   import javax.sql.DataSource;
10  
11  import com.github.davidmoten.rx.jdbc.exceptions.SQLRuntimeException;
12  
13  public final class ConnectionProviderFromContext implements ConnectionProvider {
14  
15      private final String jndiResource;
16  
17      public ConnectionProviderFromContext(String jndiResource) {
18          this.jndiResource = jndiResource;
19      }
20  
21      @Override
22      public Connection get() {
23          try {
24              Context ctx = new InitialContext();
25              DataSource ds = (DataSource) ctx.lookup(jndiResource);
26              Connection conn = ds.getConnection();
27              return conn;
28          } catch (SQLException e) {
29              throw new SQLRuntimeException(e);
30          } catch (NamingException e) {
31              throw new RuntimeException(e);
32          }
33      }
34  
35      @Override
36      public void close() {
37          // do nothing
38      }
39  
40  }
