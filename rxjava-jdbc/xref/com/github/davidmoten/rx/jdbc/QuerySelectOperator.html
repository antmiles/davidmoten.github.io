9   import rx.Observable;
10  import rx.Observable.Operator;

12  import rx.functions.Func1;

14  /**
15   * Operator corresponding to the QuerySelectOperation.
16   * 
17   * @param <T>
18   */
19  final class QuerySelectOperator<T, R> implements Operator<T, R> {
20  
21      private final Operator<T, R> operator;
22  
23      /**
24       * Constructor.
25       * 
26       * @param builder
27       * @param function
28       * @param operatorType
29       */
30      QuerySelectOperator(final QuerySelect.Builder builder, final Func1<ResultSet, T> function,
31              final OperatorType operatorType) {
32          operator = toOperator(new ApplyQuerySelect<R,T>(builder, function, operatorType));
33      }
34  
35      @Override
36      public Subscriber<? super R> call(Subscriber<? super T> subscriber) {
37          return operator.call(subscriber);
38      }
39  
40      private static class ApplyQuerySelect<R, T> implements Func1<Observable<R>, Observable<T>> {
41  
42          private Builder builder;
43          private Func1<ResultSet, T> function;
44          private OperatorType operatorType;
45  
46          private ApplyQuerySelect(QuerySelect.Builder builder, Func1<ResultSet, T> function,
47                  OperatorType operatorType) {
48              this.builder = builder;
49              this.function = function;
50              this.operatorType = operatorType;
51          }
52  
53          @Override
54          public Observable<T> call(Observable<R> observable) {
55              if (operatorType == OperatorType.PARAMETER)
56                  return builder.parameters(observable).get(function);
57              else if (operatorType == OperatorType.DEPENDENCY)
58                  // dependency
59                  return builder.dependsOn(observable).get(function);
60              else // PARAMETER_LIST
61              {
62                  @SuppressWarnings("unchecked")
63                  Observable<Observable<Object>> obs = (Observable<Observable<Object>>) observable;
64                  return obs.concatMap(new Func1<Observable<Object>, Observable<T>>() {
65                      @Override
66                      public Observable<T> call(Observable<Object> parameters) {
67                          return builder.parameters(parameters).get(function);
68                      }
69                  });
70              }
71          }
72  
73      }
74  }
