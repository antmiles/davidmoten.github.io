377     private void checkFragments(int bytesSize, int pageSize, int numRuns, Scheduler scheduler) {
378         byte[] bytes = new byte[bytesSize];
379         for (int i = 0; i < bytes.length; i++) {
380             bytes[i] = (byte) i;
381         }
382         // length field + padding field + padding + bytes = 4 + 1 + 1 + 300 =
383         // 306 bytes
384         for (int n = 1; n <= numRuns; n++) {
385             // System.out.println("bytes=" + bytesSize + " ======== " + n + "
386             // =========");
387             Flowables.repeat(bytes, n) //
388                     .compose(onBackpressureBufferToFile() //
389                             .pageSizeBytes(pageSize) //
390                             .scheduler(scheduler) //
391                             .serializerBytes()) //
392                     .doOnNext(Consumers.assertBytesEquals(bytes)) //
393                     // .doOnNext(Consumers.println()) //
394                     .doOnError(Consumers.printStackTrace()) //
395                     .test() //
396                     .awaitDone(50000000L, TimeUnit.SECONDS) //
397                     .assertValueCount(n) //
398                     .assertComplete();
399         }
400     }

426     public void testCancel() throws InterruptedException {
427         System.out.println("testCancel");
428         for (int i = 0; i < 50; i++) {
429             // System.out.println(i);
430             long n = 5000L;
431             File pageFile = new File("target/cancel" + i + ".obj");

433             byte[] bytes = new byte[40];
434             Flowables.repeat(bytes, n) //

437                             .fileFactory(Callables.constant(pageFile)) //

441                     .count() //
442                     .test() //
443                     .awaitDone(500L, TimeUnit.SECONDS) //
444                     .assertNoErrors() //

446                     .assertValue((long) 5);
447             long t = TimeUnit.SECONDS.toMillis(5);
448             while (t > 0 && pageFile.exists()) {
449                 long waitMs = 10;
450                 TimeUnit.MILLISECONDS.sleep(waitMs);
451                 t -= waitMs;
452             }
453             assertFalse(pageFile.exists());

557     @Test
558     public void testPollQueueThrowsExceptionEmitsError() {
559         PagedQueue queue = Mockito.mock(PagedQueue.class);
560         RuntimeException err = new RuntimeException();
561         Mockito.doThrow(err).when(queue).poll();
562         Worker worker = Schedulers.trampoline().createWorker();
563         TestSubscriber<String> ts = TestSubscriber.create(1);
564         BufferToFileSubscriberFlowable<String> b = new BufferToFileSubscriberFlowable<String>(ts, queue,
565                 Serializers.utf8(), worker);
566         b.onSubscribe(IGNORE);
567         b.request(1);
568         b.run();
569         Mockito.verify(queue, Mockito.atLeastOnce()).poll();
570         ts.assertError(err);
571     }
