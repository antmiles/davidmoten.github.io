13      private static final int DEFAULT_MATCH_BATCH_SIZE = 128;
14  
15      private Flowables() {
16          // prevent instantiation
17      }
18  
19      public static <A, B, K, C> Flowable<C> match(Flowable<A> a, Flowable<B> b, Function<? super A, K> aKey,
20              Function<? super B, K> bKey, BiFunction<? super A, ? super B, C> combiner, int requestSize) {
21          return new FlowableMatch<A, B, K, C>(a, b, aKey, bKey, combiner, requestSize);
22      }
23  
24      public static <A, B, K, C> Flowable<C> match(Flowable<A> a, Flowable<B> b, Function<? super A, K> aKey,
25              Function<? super B, K> bKey, BiFunction<? super A, ? super B, C> combiner) {
26          return match(a, b, aKey, bKey, combiner, DEFAULT_MATCH_BATCH_SIZE);
27      }
28  
29      public static <T> Flowable<T> repeat(T t) {
30          return new FlowableRepeat<T>(t, -1);
31      }
32  
33      public static <T> Flowable<T> repeat(T t, long count) {
34          return new FlowableRepeat<T>(t, count);
35      }
36  
37      public static <T> Flowable<T> fetchPagesByRequest(final BiFunction<? super Long, ? super Long, ? extends Flowable<T>> fetch,
38              long start, int maxConcurrent) {
39          return FlowableFetchPagesByRequest.create(fetch, start, maxConcurrent);
40      }
41  
42      public static <T> Flowable<T> fetchPagesByRequest(final BiFunction<? super Long, ? super Long, ? extends Flowable<T>> fetch,
43              long start) {
44          return fetchPagesByRequest(fetch, start, 2);
45      }
46  
47      public static <T> Flowable<T> fetchPagesByRequest(final BiFunction<? super Long, ? super Long, ? extends Flowable<T>> fetch) {
