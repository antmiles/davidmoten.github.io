78          c.setAutoCommit(true);
79          exec(c, "create table note2("
80                  + "id integer not null generated always as identity (start with 1, increment by 2),"

85      public static Database create(int maxSize, boolean big, Scheduler scheduler) {
86          NonBlockingConnectionPool pool = Pools.nonBlocking() //
87                  .connectionProvider(connectionProvider(nextUrl(), big)) //
88                  .maxPoolSize(maxSize) //
89                  .scheduler(scheduler) //
90                  .build();
91          return Database.from(pool, () -> {
92              pool.close();
93              scheduler.shutdown();
94          });
95      }

101     private static ConnectionProvider connectionProvider(String url, boolean big) {
102         return new ConnectionProvider() {
103 
104             private final AtomicBoolean once = new AtomicBoolean(false);
105             private final CountDownLatch latch = new CountDownLatch(1);
106 
107             @Override
108             public Connection get() {
109                 try {
110                     if (once.compareAndSet(false, true)) {
111                         try {
112                             Connection c = DriverManager.getConnection(url);
113                             createDatabase(c, big);
114                             return c;
115                         } finally {
116                             latch.countDown();
117                         }
118                     } else {
119                         if (latch.await(30, TimeUnit.SECONDS)) {
120                             return DriverManager.getConnection(url);
121                         } else {
122                             throw new TimeoutException("big database timed out on creation");
123                         }
124                     }
125                 } catch (SQLException | InterruptedException | TimeoutException e) {
126                     throw new SQLRuntimeException(e);
127                 }
128             }
129 
130             @Override
131             public void close() {
132                 //
133             }
134         };
135     }

145     private static void createDatabase(Connection c, boolean big) {
146         try {
147             c.setAutoCommit(true);
148             c.prepareStatement(
149                     "create table person (name varchar(50) primary key, score int not null, date_of_birth date, registered timestamp)")
150                     .execute();
151             if (big) {
152                 List<String> lines = IOUtils.readLines(
153                         DatabaseCreator.class.getResourceAsStream("/big.txt"),
154                         StandardCharsets.UTF_8);
155                 lines.stream().map(line -> line.split("\t")).forEach(items -> {
156                     try {
157                         c.prepareStatement("insert into person(name,score) values('" + items[0]
158                                 + "'," + Integer.parseInt(items[1]) + ")").execute();
159                     } catch (SQLException e) {
160                         throw new SQLRuntimeException(e);
161                     }
162                 });
163             } else {
164                 exec(c, "insert into person(name,score,registered) values('FRED',21, {ts '2015-09-17 18:47:52.69Z'})");
165                 exec(c, "insert into person(name,score) values('JOSEPH',34)");
166                 exec(c, "insert into person(name,score) values('MARMADUKE',25)");
167             }
168 
169             exec(c, "create table person_clob (name varchar(50) not null,  document clob)");
170             exec(c, "create table person_blob (name varchar(50) not null, document blob)");
171 
172             exec(c, "create table address (address_id int primary key, full_address varchar(255) not null)");
173             exec(c, "insert into address(address_id, full_address) values(1,'57 Something St, El Barrio, Big Place')");
174             exec(c, "create table note(id bigint auto_increment primary key, text varchar(255))");
175         } catch (SQLException e) {
176             throw new SQLRuntimeException(e);
177         } catch (IOException e) {
178             throw new RuntimeException(e);
179         }
180     }
