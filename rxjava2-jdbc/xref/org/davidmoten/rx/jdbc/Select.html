19  final class Select {
20  
21      private Select() {
22          // prevent instantiation
23      }
24  
25      private static final Logger log = LoggerFactory.getLogger(Select.class);
26  
27      static <T> Flowable<T> create(Single<Connection> connections,
28              Flowable<List<Object>> parameterGroups, String sql, int fetchSize,
29              Function<? super ResultSet, ? extends T> mapper, boolean eagerDispose) {
30          return connections //
31                  .toFlowable() //
32                  .flatMap(con -> create(con, sql, parameterGroups, fetchSize, mapper, eagerDispose));
33      }
34  
35      private static <T> Flowable<T> create(Connection con, String sql,
36              Flowable<List<Object>> parameterGroups, int fetchSize,
37              Function<? super ResultSet, T> mapper, boolean eagerDispose) {
38          log.debug("Select.create called with con={}", con);
39          Callable<NamedPreparedStatement> initialState = () -> Util.prepare(con, fetchSize, sql);
40          Function<NamedPreparedStatement, Flowable<T>> observableFactory = ps -> parameterGroups
41                  .flatMap(parameters -> create(con, ps.ps, parameters, mapper, ps.names), true, 1);
42          Consumer<NamedPreparedStatement> disposer = Util::closePreparedStatementAndConnection;
43          return Flowable.using(initialState, observableFactory, disposer, eagerDispose);
44      }
45  
46      private static <T> Flowable<? extends T> create(Connection con, PreparedStatement ps,
47              List<Object> parameters, Function<? super ResultSet, T> mapper, List<String> names) {
48          log.debug("parameters={}", parameters);
49          log.debug("names={}", names);
50          Callable<ResultSet> initialState = () -> Util //
51                  .setParameters(ps, parameters, names) //
52                  .executeQuery();
53          BiConsumer<ResultSet, Emitter<T>> generator = (rs, emitter) -> {
54              log.debug("getting row from ps={}, rs={}", ps, rs);
55              if (rs.next()) {
56                  T v = mapper.apply(rs);
57                  log.debug("emitting {}", v);
58                  emitter.onNext(v);
59              } else {
60                  log.debug("completed");
61                  emitter.onComplete();
62              }
63          };
64          Consumer<ResultSet> disposeState = Util::closeSilently;
65          return Flowable.generate(initialState, generator, disposeState);
66      }
