52      // represents the number of outstanding member requests.
53      // the number is decremented when a new member value is
54      // initialized (a scheduled action with a subsequent drain call)
55      // or an existing value is available from the pool (queue) (and is then
56      // emitted).
57      private final AtomicLong requested = new AtomicLong();

64      @SuppressWarnings("unchecked")
65      MemberSingle(NonBlockingPool<T> pool) {
66          Preconditions.checkNotNull(pool);
67          this.initializedAvailable = new MpscLinkedQueue<DecoratingMember<T>>();
68          this.notInitialized = new MpscLinkedQueue<DecoratingMember<T>>();
69          this.toBeReleased = new MpscLinkedQueue<DecoratingMember<T>>();
70          this.toBeChecked = new MpscLinkedQueue<DecoratingMember<T>>();
71          this.members = createMembersArray(pool.maxSize, pool.checkinDecorator);
72          for (DecoratingMember<T> m : members) {
73              notInitialized.offer(m);
74          }
75          this.scheduler = pool.scheduler;
76          this.createRetryIntervalMs = pool.createRetryIntervalMs;
77          this.observers = new AtomicReference<>(EMPTY);
78          this.pool = pool;
79      }

81      private DecoratingMember<T>[] createMembersArray(int poolMaxSize,
82              BiFunction<? super T, ? super Checkin, ? extends T> checkinDecorator) {
83          @SuppressWarnings("unchecked")
84          DecoratingMember<T>[] m = new DecoratingMember[poolMaxSize];
85          for (int i = 0; i < m.length; i++) {
86              m[i] = new DecoratingMember<T>(null, checkinDecorator, this);
87          }
88          return m;
89      }

91      @Override
92      protected void subscribeActual(SingleObserver<? super Member<T>> observer) {
93          // the action of checking out a member from the pool is implemented as a
94          // subscription to the singleton MemberSingle
95          MemberSingleObserver<T> md = new MemberSingleObserver<T>(observer, this);
96          observer.onSubscribe(md);
97          if (pool.isClosed()) {
98              observer.onError(new PoolClosedException());
99              return;
100         }
101         add(md);
102         if (md.isDisposed()) {
103             remove(md);
104         }
105         requested.incrementAndGet();
106         log.debug("subscribed");
107         drain();
108     }

110     public void checkin(Member<T> member) {
111         checkin(member, false);
112     }

114     public void checkin(Member<T> member, boolean decrementInitializeScheduled) {
115         log.debug("checking in {}", member);
116         DecoratingMember<T> d = ((DecoratingMember<T>) member);
117         d.scheduleRelease();
118         d.markAsChecked();
119         initializedAvailable.offer((DecoratingMember<T>) member);
120         if (decrementInitializeScheduled) {
121             initializeScheduled.decrementAndGet();
122         }
123         drain();
124     }

127         toBeReleased.offer(member);
128         drain();
129     }

152     private void drain() {
153         log.debug("drain called");
154         if (wip.getAndIncrement() == 0) {
155             log.debug("drain loop starting");
156             int missed = 1;
157             while (true) {
158                 // we schedule release of members even if no requests exist
159                 scheduleReleases();
160                 scheduleChecks();
161 
162                 long r = requested.get(); // requested
163                 log.debug("requested={}", r);
164 
165                 long e = 0; // emitted
166                 while (e != r) {
167                     if (cancelled) {
168                         disposeAll();
169                         return;
170                     }
171                     Observers<T> obs = observers.get();
172                     // the check below is required so a tryEmit that returns false doesn't bring
173                     // abouts a spin on this loop
174                     int c = obs.activeCount;
175                     // if there have been some cancellations then adjust the requested amount by
176                     // increasing emitted e
177                     e += Math.max(0, r - e - c);
178                     if (c == 0) {
179                         // if no observers then we break the loop
180                         break;
181                     }
182                     // check for an already initialized available member
183                     final DecoratingMember<T> m = initializedAvailable.poll();
184                     log.debug("poll of available members returns {}", m);
185                     if (m == null) {
186                         // no members available, check for a released member (that needs to be
187                         // reinitialized before use)
188                         final DecoratingMember<T> m2 = notInitialized.poll();
189                         if (m2 == null) {
190                             break;
191                         } else {
192                             // only schedule member initialization if there is enough demand,
193                             boolean used = trySchedulingInitialization(r, e, m2);
194                             if (!used) {
195                                 break;
196                             }
197                         }
198                     } else if (!m.isReleasing() && !m.isChecking()) {
199                         log.debug("trying to emit member");
200                         if (shouldPerformHealthCheck(m)) {
201                             log.debug("queueing member for health check {}", m);
202                             toBeChecked.offer(m);
203                         } else {
204                             log.debug("no health check required for {}", m);
205                             // this should not block because it just schedules emissions to observers
206                             if (tryEmit(obs, m)) {
207                                 e++;
208                             } else {
209                                 log.debug("no active observers");
210                             }
211                         }
212                     }
213                     // schedule release immediately of any member
214                     // queued for releasing
215                     scheduleReleases();
216                     // schedule check of any member queued for checking
217                     scheduleChecks();
218 
219                 }
220                 // normally we don't reduce requested if it is Long.MAX_VALUE
221                 // but given that the only way to increase requested is by subscribing
222                 // (which increases it by one only) then requested will never be Long.MAX_VALUE
223                 if (e != 0L) {
224                     requested.addAndGet(-e);
225                 }
226                 missed = wip.addAndGet(-missed);
227                 if (missed == 0) {
228                     return;
229                 }
230             }
231         }
232     }

234     private boolean trySchedulingInitialization(long r, long e, final DecoratingMember<T> m) {
235         // check initializeScheduled using a CAS loop
236         while (true) {
237             long cs = initializeScheduled.get();
238             if (e + cs < r) {
239                 if (initializeScheduled.compareAndSet(cs, cs + 1)) {
240                     log.debug("scheduling member creation");
241                     scheduled.add(scheduler.scheduleDirect(new Initializer(m)));
242                     return true;
243                 }
244             } else {
245                 log.debug("insufficient demand to initialize {}", m);
246                 // don't need to initialize more so put back on queue and exit the loop
247                 notInitialized.offer(m);
248                 return false;
249             }
250         }
251     }
252 
253     private boolean shouldPerformHealthCheck(final DecoratingMember<T> m) {
254         long now = scheduler.now(TimeUnit.MILLISECONDS);
255         log.debug("schedule.now={}, lastCheck={}", now, m.lastCheckTime());
256         return pool.idleTimeBeforeHealthCheckMs > 0 && now - m.lastCheckTime() >= pool.idleTimeBeforeHealthCheckMs;
257     }
258 
259     private void scheduleChecks() {
260         DecoratingMember<T> m;
261         while ((m = toBeChecked.poll()) != null) {
262             if (!m.isReleasing()) {
263                 log.debug("scheduling check of {}", m);
264                 // we mark as checking so that we can ignore it if already in the
265                 // initializedAvailable queue after concurrent checkin
266                 m.markAsChecking();
267                 scheduled.add(scheduler.scheduleDirect(new Checker(m)));
268             }
269         }
270     }
271 
272     private void scheduleReleases() {
273         DecoratingMember<T> m;
274         while ((m = toBeReleased.poll()) != null) {
275             log.debug("scheduling release of {}", m);
276             // we mark as releasing so that we can ignore it if already in the
277             // initializedAvailable queue after concurrent checkin
278             m.markAsReleasing();
279             scheduled.add(scheduler.scheduleDirect(new Releaser(m)));
280         }
281     }

283     private boolean tryEmit(Observers<T> obs, DecoratingMember<T> m) {
284         // get a fresh worker each time so we jump threads to
285         // break the stack-trace (a long-enough chain of
286         // checkout-checkins could otherwise provoke stack
287         // overflow)
288 
289         // advance counter so the next and choose an Observer to emit to (round robin)
290 
291         int index = obs.index;
292         MemberSingleObserver<T> o = obs.observers[index];
293         MemberSingleObserver<T> oNext = o;
294         // atomically bump up the index (if that entry has not been deleted in
295         // the meantime by disposal)
296         while (true) {
297             Observers<T> x = observers.get();
298             if (x.index == index && x.observers[index] == o) {
299                 boolean[] active = new boolean[x.active.length];
300                 System.arraycopy(x.active, 0, active, 0, active.length);
301                 int nextIndex = (index + 1) % active.length;
302                 while (nextIndex != index && !active[nextIndex]) {
303                     nextIndex = (nextIndex + 1) % active.length;
304                 }
305                 active[nextIndex] = false;
306                 if (observers.compareAndSet(x, new Observers<T>(x.observers, active, x.activeCount - 1, nextIndex))) {
307                     oNext = x.observers[nextIndex];
308                     break;
309                 }
310             } else {
311                 // checkin because no active observers
312                 m.checkin();
313                 return false;
314             }
315         }
316         Worker worker = scheduler.createWorker();
317         worker.schedule(new Emitter<T>(worker, oNext, m));
318         return true;
319     }

321     @VisibleForTesting
322     final class Initializer implements Runnable {
323 
324         private final DecoratingMember<T> m;
325 
326         Initializer(DecoratingMember<T> m) {
327             this.m = m;
328         }
329 
330         @Override
331         public void run() {
332             if (!cancelled) {
333                 try {
334                     log.debug("creating value");
335                     // this action might block so is scheduled
336                     T value = pool.factory.call();
337                     m.setValueAndClearReleasingFlag(value);
338                     requested.incrementAndGet();
339                     checkin(m, true);
340                 } catch (Throwable t) {
341                     RxJavaPlugins.onError(t);
342                     // check cancelled again because factory.call() is user specified and could have
343                     // taken a significant time to complete
344                     if (!cancelled) {
345                         // schedule a retry
346                         scheduled.add(scheduler.scheduleDirect(this, createRetryIntervalMs, TimeUnit.MILLISECONDS));
347                     }
348                 }
349             }
350         }
351     }

372     final class Checker implements Runnable {
373 
374         private final DecoratingMember<T> m;
375 
376         Checker(DecoratingMember<T> m) {
377             this.m = m;
378         }
379 
380         @Override
381         public void run() {
382             try {
383                 log.debug("performing health check on {}", m);
384                 if (!pool.healthCheck.test(m.value())) {
385                     log.debug("failed health check");
386                     m.disposeValue();
387                     log.debug("scheduling recreation of member {}", m);
388                     scheduled.add(scheduler.scheduleDirect(() -> {
389                         log.debug("recreating member after failed health check {}", m);
390                         notInitialized.offer(m);
391                         drain();
392                     }, pool.createRetryIntervalMs, TimeUnit.MILLISECONDS));
393                 } else {
394                     m.markAsChecked();
395                     initializedAvailable.offer(m);
396                     drain();
397                 }
398             } catch (Throwable t) {
399                 RxJavaPlugins.onError(t);
400             }
401         }
402     }

424     void add(@NonNull MemberSingleObserver<T> inner) {
425         while (true) {
426             Observers<T> a = observers.get();
427             int n = a.observers.length;
428             @SuppressWarnings("unchecked")
429             MemberSingleObserver<T>[] b = new MemberSingleObserver[n + 1];
430             System.arraycopy(a.observers, 0, b, 0, n);
431             b[n] = inner;
432             boolean[] active = new boolean[n + 1];
433             System.arraycopy(a.active, 0, active, 0, n);
434             active[n] = true;
435             if (observers.compareAndSet(a, new Observers<T>(b, active, a.activeCount + 1, a.index))) {
436                 return;
437             }
438         }
439     }

441     @SuppressWarnings("unchecked")
442     private void removeAllObservers() {
443         while (true) {
444             Observers<T> a = observers.get();
445             if (observers.compareAndSet(a, EMPTY)) {
446                 return;
447             }
448         }
449     }

451     @SuppressWarnings("unchecked")
452     void remove(@NonNull MemberSingleObserver<T> inner) {
453         while (true) {
454             Observers<T> a = observers.get();
455             int n = a.observers.length;
456             if (n == 0) {
457                 return;
458             }
459 
460             int j = -1;
461 
462             for (int i = 0; i < n; i++) {
463                 if (a.observers[i] == inner) {
464                     j = i;
465                     break;
466                 }
467             }
468 
469             if (j < 0) {
470                 return;
471             }
472             Observers<T> next;
473             if (n == 1) {
474                 next = EMPTY;
475             } else {
476                 MemberSingleObserver<T>[] b = new MemberSingleObserver[n - 1];
477                 System.arraycopy(a.observers, 0, b, 0, j);
478                 System.arraycopy(a.observers, j + 1, b, j, n - j - 1);
479                 boolean[] active = new boolean[n - 1];
480                 System.arraycopy(a.active, 0, active, 0, j);
481                 System.arraycopy(a.active, j + 1, active, j, n - j - 1);
482                 int nextActiveCount = a.active[j] ? a.activeCount - 1 : a.activeCount;
483                 if (a.index >= j && a.index > 0) {
484                     next = new Observers<T>(b, active, nextActiveCount, a.index - 1);
485                 } else {
486                     next = new Observers<T>(b, active, nextActiveCount, a.index);
487                 }
488             }
489             if (observers.compareAndSet(a, next)) {
490                 return;
491             }
492         }
493     }

495     private static final class Observers<T> {
496         final MemberSingleObserver<T>[] observers;
497         // an observer is active until it is emitted to
498         final boolean[] active;
499         final int activeCount;
500         final int index;
501 
502         Observers(MemberSingleObserver<T>[] observers, boolean[] active, int activeCount, int index) {
503             Preconditions.checkArgument(observers.length > 0 || index == 0, "index must be -1 for zero length array");
504             Preconditions.checkArgument(observers.length == active.length);
505             this.observers = observers;
506             this.index = index;
507             this.active = active;
508             this.activeCount = activeCount;
509         }
510     }

512     private static final class Emitter<T> implements Runnable {
513 
514         private final Worker worker;
515         private final MemberSingleObserver<T> observer;
516         private final Member<T> m;
517 
518         Emitter(Worker worker, MemberSingleObserver<T> observer, Member<T> m) {
519             this.worker = worker;
520             this.observer = observer;
521             this.m = m;
522         }
523 
524         @Override
525         public void run() {
526             worker.dispose();
527             try {
528                 observer.child.onSuccess(m);
529             } catch (Throwable e) {
530                 RxJavaPlugins.onError(e);
531             } finally {
532                 observer.dispose();
533             }
534         }
535     }

537     public void release(DecoratingMember<T> m) {

543     static final class MemberSingleObserver<T> extends AtomicReference<MemberSingle<T>> implements Disposable {
544 
545         private static final long serialVersionUID = -7650903191002190468L;
546 
547         final SingleObserver<? super Member<T>> child;
548 
549         MemberSingleObserver(SingleObserver<? super Member<T>> child, MemberSingle<T> parent) {
550             this.child = child;
551             lazySet(parent);
552         }
553 
554         @Override
555         public void dispose() {
556             MemberSingle<T> parent = getAndSet(null);
557             if (parent != null) {
558                 parent.remove(this);
559             }
560         }
561 
562         @Override
563         public boolean isDisposed() {
564             return get() == null;
565         }
