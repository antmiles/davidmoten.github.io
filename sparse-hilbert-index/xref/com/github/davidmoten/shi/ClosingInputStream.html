1   package com.github.davidmoten.shi;

3   import java.io.IOException;
4   import java.io.InputStream;
5   
6   import io.reactivex.functions.Action;
7   
8   class ClosingInputStream extends InputStream {
9   
10      private final InputStream in;
11      private final Action action;
12  
13      ClosingInputStream(InputStream in, Action action) {
14          this.in = in;
15          this.action = action;
16      }
17  
18      @Override
19      public int read() throws IOException {
20          return in.read();
21      }
22  
23      @Override
24      public int read(byte[] b, int off, int len) throws IOException {
25          return in.read(b, off, len);
26      }
27  
28      @Override
29      public void close() throws IOException {
30          try {
31              in.close();
32          } finally {
33              try {
34                  action.run();
35              } catch (Exception e) {
36                  throw new RuntimeException(e);
37              }
38          }
39      }
40  }
