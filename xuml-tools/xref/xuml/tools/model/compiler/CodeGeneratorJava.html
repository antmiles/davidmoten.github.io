72  	private void generateEntitySources() {
73  		log("generating " + entitySourceDirectory);
74  		ModeledDomain md = domain;
75  		Lookups lookups = new Lookups(domains, md);
76  		for (Class cls : getClasses(md)) {
77  			createEntityJavaSource(cls, entitySourceDirectory, lookups);
78  //			createImplementationJavaSource(cls, implementationSourceDirectory,
79  //					lookups);
80  		}
81  		if (generatePersistenceXml)
82  			createPersistenceXml(domain, new File(resourcesDirectory,
83  					"META-INF/persistence.xml"));
84  		createContext(domain, entitySourceDirectory, lookups);
85  		log("finished generation");
86  	}

88  	private void createImplementationJavaSource(Class cls, File destination,
89  			Lookups lookups) {
90  		ClassInfo info = createClassInfo(cls);
91  		if (info.hasBehaviour()) {
92  			log("generating " + getFullClassImplementationName(cls));
93  			BehaviourImplementationWriter w = new BehaviourImplementationWriter(
94  					info, getFullClassImplementationName(cls));
95  			String java = w.generate();
96  			File file = new File(destination,
97  					getClassImplementationFilename(cls));
98  			if (!file.exists() || overwriteImplementation)
99  				writeToFile(java.getBytes(), file);
100 		}
101 	}

103 	private String getClassImplementationFilename(Class cls) {
104 		String s = getFullClassImplementationName(cls);
105 		return s.replace(".", "/") + ".java";
106 	}

108 	private String getFullClassImplementationName(Class cls) {
109 		return implementationPackageName + "." + getClassJavaSimpleName(cls)

113 	private void createPersistenceXml(ModeledDomain domain, File file) {
114 		OutputStream out;
115 		try {
116 			file.getParentFile().mkdirs();
117 			out = new FileOutputStream(file);
118 			String xml = generatePersistenceXml(domain);
119 			out.write(xml.toString().getBytes());
120 			out.close();
121 		} catch (IOException e) {
122 			throw new RuntimeException(e);
123 		}
124 	}

126 	private String generatePersistenceXml(ModeledDomain domain) {
127 		List<String> classes = Lists.newArrayList();
128 		for (Class cls : getClasses(domain)) {
129 			ClassInfo info = createClassInfo(cls);
130 			classes.add(info.getClassFullName());
131 		}
132 		String xml = new PersistenceXmlWriter().generate(classes);
133 		return xml;
134 	}

136 	private List<Class> getClasses(ModeledDomain domain) {
137 		List<Class> list = Lists.newArrayList();
138 		for (Subsystem subsystem : domain.getSubsystem()) {
139 			for (JAXBElement<? extends SubsystemElement> element : subsystem
140 					.getSubsystemElement()) {
141 				if (element.getValue() instanceof Class) {
142 					Class cls = (Class) element.getValue();
143 					list.add(cls);
144 				}
145 			}
146 		}
147 		return list;
148 	}

150 	private void createContext(ModeledDomain domain, File destination,
151 			Lookups lookups) {
152 		ByteArrayOutputStream bytes = new ByteArrayOutputStream();
153 		PrintStream out = new PrintStream(bytes);
154 
155 		TypeRegister types = new TypeRegister();
156 		out.format("public class Context {\n\n");
157 		out.format("    private static %s signaller;\n\n",
158 				types.addType(Signaller.class));
159 		out.format("    public static int sendSignalsInQueue() {\n");
160 		out.format("        return signaller.sendSignalsInQueue();\n");
161 		out.format("    }\n\n");
162 		out.format(
163 				"    public static <T extends %s<T>> long persistSignal(String fromEntityUniqueId, Object id, Class<T> cls, %s<T> event, long time, %s<Long> repeatIntervalMs) {\n",
164 				types.addType(Entity.class), types.addType(Event.class), types.addType(Optional.class));
165 		out.format("        return signaller.persistSignal(fromEntityUniqueId, id,cls,event,time,repeatIntervalMs);\n");
166 		out.format("    }\n\n");
167 		out.format("    public static void stop() {\n");
168 		out.format("        signaller.stop();\n");
169 		out.format("    }\n\n");
170 		out.format(
171 				"    public static <T extends %s<T>> T create(%s<T> cls, %s<T> event) {\n",
172 				types.addType(Entity.class),
173 				types.addType(java.lang.Class.class),
174 				types.addType(CreationEvent.class));
175 		out.format("        return signaller.create(cls,event);\n");
176 		out.format("    }\n\n");
177 		out.format(
178 				"    public static void setEntityManagerFactory(%s emf) {\n",
179 				types.addType(EntityManagerFactory.class));
180 		out.format("        signaller = new %s(emf,listenerFactory);\n",
181 				types.addType(Signaller.class), types.addType(Signaller.class));
182 		for (Subsystem subsystem : domain.getSubsystem()) {
183 			for (JAXBElement<? extends SubsystemElement> element : subsystem
184 					.getSubsystemElement()) {
185 				if (element.getValue() instanceof Class) {
186 					Class cls = (Class) element.getValue();
187 					// create classes (impls)
188 					ClassInfo info = createClassInfo(cls);
189 					if (info.hasBehaviour())
190 						out.format("        %s.setSignaller_(signaller);\n",
191 								types.addType(info.getClassFullName()));
192 				}
193 			}
194 		}
195 		out.format("    }\n\n");
196 
197 		out.format("    private static %s listenerFactory;\n\n",
198 				types.addType(SignalProcessorListenerFactory.class));
199 		out.format(
200 				"    public static void setEntityActorListenerFactory(%s listenerFactory) {\n",
201 				types.addType(SignalProcessorListenerFactory.class));
202 		out.format("        if (signaller !=null)\n");
203 		out.format(
204 				"            throw new %s(\"EntityActorListenerFactory must be set before EntityManagerFactory\");\n",
205 				types.addType(RuntimeException.class));
206 		out.format("        Context.listenerFactory = listenerFactory;\n");
207 		out.format("    }\n\n");
208 
209 		out.format("    public static %s createEntityManager() {\n",
210 				types.addType(EntityManager.class));
211 		out.format("        return signaller.getEntityManagerFactory().createEntityManager();\n");
212 		out.format("    }\n\n");
213 		out.format("    public static void close() {\n");
214 		out.format("        signaller.close();\n");
215 		out.format("    }\n\n");
216 
217 		out.format(
218 				"    public static <T extends %s<T>> T remove(T entity) {\n",
219 				types.addType(Entity.class));
220 		out.format("        boolean emOpenAlready = em()!=null;\n");
221 		out.format("        %s em;\n", types.addType(EntityManager.class));
222 		out.format("        if (emOpenAlready)\n");
223 		out.format("            em = em();\n");
224 		out.format("        else\n");
225 		out.format("            em = createEntityManager();\n");
226 		out.format("        em.remove(entity);\n");
227 		out.format("        if (!emOpenAlready)\n");
228 		out.format("            em.close();\n");
229 		out.format("        return entity;\n");
230 		out.format("    }\n\n");
231 
232 		out.format("    public static <T extends %s<T>> T load(T entity) {\n",
233 				types.addType(Entity.class));
234 		out.format("        boolean emOpenAlready = em()!=null;\n");
235 		out.format("        %s em;\n", types.addType(EntityManager.class));
236 		out.format("        if (emOpenAlready)\n");
237 		out.format("            em = em();\n");
238 		out.format("        else\n");
239 		out.format("            em = createEntityManager();\n");
240 		out.format("        T t = em.merge(entity);\n");
241 		out.format("        em.refresh(t);\n");
242 		out.format("        if (!emOpenAlready)\n");
243 		out.format("            em.close();\n");
244 		out.format("        return t;\n");
245 		out.format("    }\n\n");
246 
247 		out.format("    public static %s em() {\n",
248 				types.addType(EntityManager.class));
249 		out.format("        return signaller.getInfo().getCurrentEntityManager();\n");
250 		out.format("    }\n\n");
251 
252 		out.format("}");
253 		out.close();
254 
255 		String s = "package " + domainPackageName + ";\n\n";
256 		s += types.getImports(domainPackageName + ".Context") + "\n";
257 		s += bytes.toString();
258 
259 		String filename = domainPackageName.replace(".", "/") + "/Context.java";
260 		try {
261 			FileOutputStream fos = new FileOutputStream(new File(destination,
262 					filename));
263 			fos.write(s.getBytes());
264 			fos.close();
265 		} catch (IOException e) {
266 			throw new RuntimeException(e);
267 		}
268 	}

274 	private void createEntityJavaSource(Class cls, File destination,
275 			Lookups lookups) {
276 		ClassWriter w = new ClassWriter(createClassInfo(cls));
277 		String java = w.generate();
278 		File file = new File(destination, getClassFilename(cls));
279 		writeToFile(java.getBytes(), file);
280 	}

282 	private ClassInfo createClassInfo(Class cls) {
283 		Lookups lookups = new Lookups(domains, domain);
284 		return new ClassInfo(nameManager, cls, domainPackageName,
285 				"description", domainSchema, lookups);
286 	}

288 	private String getClassJavaSimpleName(Class cls) {
289 		return cls.getName().replace(" ", "").replace("-", "");
290 	}

292 	private String getFullClassName(Class cls) {
293 		return domainPackageName + "." + getClassJavaSimpleName(cls);
294 	}

297 		String s = getFullClassName(cls);

306 		try {
307 			file.getParentFile().mkdirs();
308 			log("writing to " + file);
309 			FileOutputStream fos = new FileOutputStream(file);
310 			fos.write(bytes);
311 			fos.close();

313 			throw new RuntimeException(e);
314 		}
